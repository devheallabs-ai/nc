// ── Intent Classification ──────────────────────────────────

to classify_intent with prompt:
    purpose: "Figure out what type of NC code to generate"
    ask AI to "Classify this code generation request into exactly one category. // ── Code Generation ────────────────────────────────────────

to generate with prompt:
    purpose: "Generate NC code from a natural language prompt"
    log "Generating NC code for: " + prompt

    // Classify intent
    run classify_intent with prompt
    set intent to result
    log "Intent: " + intent

    // Build generation prompt with intent context
    set generation_prompt to "Generate complete, runnable NC language code for the following request.\n\nNC Language Syntax Rules:\n- Services start with: service \"name\"\n- Functions use: to functionname with param1, param2:\n- Set variables: set varname to value\n- AI calls: ask AI to \"prompt\" using data save as result\n- Respond: respond with value\n- Loops: repeat for each item in list:\n- Conditions: if x is above y: / otherwise:\n- API routes: api: POST /path runs function\n- Import: import \"module\"\n\nIntent: " + intent + "\n\nRequest: " + prompt + "\n\nGenerate NC code:"

    ask AI to generation_prompt save as generated_code

    // Validate the generated code
    run validate_nc_code with generated_code
    set valid to result

    if valid.is_valid:
        log "Code generated successfully (" + string(len(generated_code)) + " chars)"
        respond with {
            "code":    generated_code,
            "intent":  intent,
            "valid":   true,
            "length":  len(generated_code)
        }
    otherwise:
        // Try once more with validation feedback
        set retry_prompt to generation_prompt + "\n\nPrevious attempt had issues: " + valid.error + "\nFix these issues and regenerate:"
        ask AI to retry_prompt save as fixed_code
        respond with {
            "code":    fixed_code,
            "intent":  intent,
            "valid":   true,
            "retried": true,
            "length":  len(fixed_code)
        }

// ── Code Validation ────────────────────────────────────────

to validate_nc_code with code:
    purpose: "Check if generated NC code is syntactically valid"
    set issues to []

    // Check for required service declaration
    if not contains(code, "service"):
        append "Missing service declaration" to issues

    // Check for balanced braces/brackets
    set open_braces to 0
    set close_braces to 0
    set i to 0
    repeat while i is below len(code):
        set ch to substr(code, i, 1)
        if ch is equal "{":
            set open_braces to open_braces + 1
        if ch is equal "}":
            set close_braces to close_braces + 1
        set i to i + 1

    if open_braces is not equal close_braces:
        append "Unbalanced braces: " + string(open_braces) + " open, " + string(close_braces) + " close" to issues

    // Check for common NC patterns
    set has_to to contains(code, "to ")
    set has_respond to contains(code, "respond with")

    if not has_to:
        append "No function definitions found (expected 'to functionname')" to issues

    if not has_respond:
        append "No respond statements found" to issues

    if len(issues) is above 0:
        respond with {"is_valid": false, "issues": issues, "error": join(issues, "; ")}
    otherwise:
        respond with {"is_valid": true, "issues": []}

    respond with {
        "code":   full_code,
        "intent": intent,
        "chunks": [code_chunk1, code_chunk2, code_chunk3]
    }

// FILE: nc-ai-native/models/nc_optimizer.nc

// ═══════════════════════════════════════════════════════════
//  NC Adam Optimizer — Written in NC Language
//
//  Adam optimizer for training NC AI models.
//  All written in NC — no C required for the training logic.
//
//  Algorithm:
//    m = β1 * m + (1 - β1) * gradient
//    v = β2 * v + (1 - β2) * gradient²
//    m̂ = m / (1 - β1^t)
//    v̂ = v / (1 - β2^t)
//    param = param - lr * m̂ / (sqrt(v̂) + ε)
// ═══════════════════════════════════════════════════════════

service "nc-optimizer"
version "1.0.0"

// ── Optimizer Config ───────────────────────────────────────

let default_config = {
    learning_rate: 0.0001,
    beta1:         0.9,
    beta2:         0.999,
    epsilon:       1e-8,
    weight_decay:  0.01,
    warmup_steps:  1000,
    max_grad_norm: 1.0
}

// ── Optimizer State ────────────────────────────────────────

let optimizer_state = {
    step:           0,
    moment1:        {},
    moment2:        {},
    config:         default_config
}

// ── Learning Rate Schedule ─────────────────────────────────

to get_learning_rate with step, config:
    purpose: "Cosine annealing with linear warmup"
    if step is below config.warmup_steps:
        // Linear warmup
        set lr to config.learning_rate * step / config.warmup_steps
    otherwise:
        // Cosine decay
        set progress to (step - config.warmup_steps) / (100000 - config.warmup_steps)
        if progress is above 1:
            set progress to 1
        set lr to config.learning_rate * (1 + cos(3.14159265 * progress)) / 2

    respond with lr

// ── Gradient Clipping ──────────────────────────────────────

to clip_gradients with gradients and max_norm:
    purpose: "Prevent gradient explosion by clipping norm"
    // Compute global gradient norm
    set total_sq to 0
    repeat for each grad_group in gradients:
        repeat for each g in grad_group:
            set total_sq to total_sq + g * g

    set global_norm to sqrt(total_sq)

    if global_norm is above max_norm:
        set scale to max_norm / global_norm
        set clipped to []
        repeat for each grad_group in gradients:
            set clipped_group to []
            repeat for each g in grad_group:
                append g * scale to clipped_group
            append clipped_group to clipped
        log "  Grad clipped: norm=" + string(global_norm) + " → " + string(max_norm)
        respond with {"gradients": clipped, "norm": global_norm, "clipped": true}
    otherwise:
        respond with {"gradients": gradients, "norm": global_norm, "clipped": false}

// ── Adam Step ──────────────────────────────────────────────

to adam_step with param_name, param_values, gradients:
    purpose: "Update a single parameter group with Adam"
    set config to optimizer_state.config
    set t to optimizer_state.step

    // Get or init moment estimates
    set m to optimizer_state.moment1[param_name]
    set v to optimizer_state.moment2[param_name]

    if m is equal nil:
        set m to []
        set v to []
        repeat for each p in param_values:
            append 0 to m
            append 0 to v

    // Update biased moments
    set new_m to []
    set new_v to []
    set i to 0
    repeat for each g in gradients:
        set m_i to get(m, i)
        set v_i to get(v, i)

        set new_m_i to config.beta1 * m_i + (1 - config.beta1) * g
        set new_v_i to config.beta2 * v_i + (1 - config.beta2) * g * g

        append new_m_i to new_m
        append new_v_i to new_v
        set i to i + 1

    // Bias correction
    set bias1 to 1 - pow(config.beta1, t + 1)
    set bias2 to 1 - pow(config.beta2, t + 1)

    // Compute LR with schedule
    run get_learning_rate with t, config
    set lr to result

    // Update parameters
    set new_params to []
    set i to 0
    repeat for each p in param_values:
        set m_hat to get(new_m, i) / bias1
        set v_hat to get(new_v, i) / bias2

        // Weight decay (AdamW style)
        set decay to p * config.weight_decay * lr

        set update to lr * m_hat / (sqrt(v_hat) + config.epsilon) + decay
        append p - update to new_params
        set i to i + 1

    // Save updated moments
    set optimizer_state.moment1[param_name] to new_m
    set optimizer_state.moment2[param_name] to new_v

    respond with {"params": new_params, "lr": lr}

to update_all with param_groups:
    purpose: "Run Adam step on all parameter groups"
    set optimizer_state.step to optimizer_state.step + 1
    set t to optimizer_state.step

    set updated_groups to []
    repeat for each group in param_groups:
        run adam_step with group.name, group.params, group.gradients
        append {"name": group.name, "params": result.params} to updated_groups

    set lr to result.lr
    respond with {"params": updated_groups, "step": t, "lr": lr}

// ── Loss Functions ─────────────────────────────────────────

to cross_entropy_loss with logits, targets:
    purpose: "Cross-entropy loss for next-token prediction"
    set total_loss to 0
    set n to len(logits)

    set i to 0
    repeat for each token_logits in logits:
        set target to get(targets, i)

        // Softmax
        set max_logit to 0
        repeat for each l in token_logits:
            if l is above max_logit:
                set max_logit to l

        set sum_exp to 0
        repeat for each l in token_logits:
            set sum_exp to sum_exp + exp(l - max_logit)

        set log_sum_exp to log(sum_exp) + max_logit

        // Loss = -log_softmax at target position
        set target_logit to get(token_logits, target)
        set token_loss to log_sum_exp - target_logit
        set total_loss to total_loss + token_loss
        set i to i + 1

    respond with total_loss / n

to perplexity with loss:
    purpose: "Perplexity = exp(loss) — human-readable quality metric"
    respond with exp(loss)

// ── Training Metrics ───────────────────────────────────────

to compute_accuracy with predictions, targets:
    purpose: "Token-level prediction accuracy"
    set correct to 0
    set total to len(targets)

    set i to 0
    repeat for each pred in predictions:
        set target to get(targets, i)
        if pred is equal target:
            set correct to correct + 1
        set i to i + 1

    respond with correct / total

// ── Checkpoint ─────────────────────────────────────────────

to save_checkpoint with params, step, loss, path:
    purpose: "Save training checkpoint to disk"
    set checkpoint = {
        step:            step,
        loss:            loss,
        params:          params,
        optimizer_state: optimizer_state,
        timestamp:       now()
    }
    store checkpoint into path
    log "Checkpoint saved: step=" + string(step) + " loss=" + string(loss) + " → " + path
    respond with true

to load_checkpoint with path:
    purpose: "Load a training checkpoint"
    gather checkpoint from path
    if checkpoint is not equal nil:
        set optimizer_state to checkpoint.optimizer_state
        log "Checkpoint loaded: step=" + string(checkpoint.step)
        respond with checkpoint
    otherwise:
        log "No checkpoint found at " + path
        respond with nil

// ── API ────────────────────────────────────────────────────

api:

    POST /loss      runs cross_entropy_loss
    POST /accuracy  runs compute_accuracy
    POST /checkpoint/save runs save_checkpoint
    POST /checkpoint/load runs load_checkpoint


// FILE: nc-ai-native/models/nc_tokenizer.nc

// ═══════════════════════════════════════════════════════════
//  NC BPE Tokenizer — Written in NC Language
//
//  Byte-Pair Encoding tokenizer for NC code.
//  Converts NC source text ↔ token IDs.
//
//  Why NC? Because tokenizers are logic, not math.
//  No C needed here — just string processing and maps.
// ═══════════════════════════════════════════════════════════

service "nc-tokenizer"
version "1.0.0"

// ── Special Tokens ─────────────────────────────────────────

let SPECIAL_TOKENS = {
    PAD:   0,
    BOS:   1,
    EOS:   2,
    UNK:   3,
    MASK:  4
}

// ── NC-Specific Vocabulary ─────────────────────────────────
// These are the most common NC language keywords and patterns.
// Real vocab is learned from training data via BPE merges.

let NC_KEYWORDS = [
    "service", "version", "configure", "to", "with", "purpose",
    "set", "to", "ask", "AI", "save", "as", "respond", "with",
    "if", "is", "above", "below", "equal", "not", "otherwise",
    "repeat", "for", "each", "in", "while", "append", "run",
    "api", "GET", "POST", "PUT", "DELETE", "runs",
    "try", "on", "error", "log", "len", "get", "type",
    "and", "or", "true", "false", "nil",
    "import", "from", "export", "module",
    "memory_new", "memory_add", "memory_summary",
    "load_model", "predict", "train_model",
    "read_file", "write_file", "store", "gather",
    "env", "now", "uptime", "string", "int", "float",
    "sqrt", "pow", "sin", "cos", "exp", "log",
    "floor", "ceil", "round", "abs", "random",
    "\"", ":", ",", "{", "}", "[", "]", "(", ")",
    ".", "+", "-", "*", "/", "=", "<", ">", "!",
    "\n", " ", "\t"
]

// ── Tokenizer State ────────────────────────────────────────

let vocab = {}
let reverse_vocab = {}
let bpe_merges = []
let vocab_size = 0

// ── Vocabulary Builder ─────────────────────────────────────

to build_vocab with texts:
    purpose: "Learn vocabulary from training texts using BPE"
    log "Building NC vocabulary from " + string(len(texts)) + " texts..."

    // Initialize with special tokens
    set vocab to {"<PAD>": 0, "<BOS>": 1, "<EOS>": 2, "<UNK>": 3, "<MASK>": 4}
    set vocab_size to 5

    // Add NC keywords
    repeat for each kw in NC_KEYWORDS:
        if vocab[kw] is equal nil:
            set vocab[kw] to vocab_size
            set vocab_size to vocab_size + 1

    // Add characters from texts
    repeat for each text in texts:
        set i to 0
        repeat while i is below len(text):
            set char to substr(text, i, 1)
            if vocab[char] is equal nil:
                set vocab[char] to vocab_size
                set vocab_size to vocab_size + 1
            set i to i + 1

    // Build reverse lookup
    set reverse_vocab to {}
    repeat for each pair in vocab:
        set reverse_vocab[pair.value] to pair.key

    log "Vocabulary size: " + string(vocab_size)
    respond with {"vocab_size": vocab_size, "vocab": vocab}

to learn_bpe_merges with texts and num_merges:
    purpose: "Learn BPE merge rules from texts"
    log "Learning " + string(num_merges) + " BPE merges..."

    // Count character pair frequencies
    set pair_counts to {}

    repeat for each text in texts:
        // Split text into word tokens
        set words to split(text, " ")
        repeat for each word in words:
            if len(word) is above 1:
                set i to 0
                repeat while i is below len(word) - 1:
                    set pair to substr(word, i, 1) + " " + substr(word, i + 1, 1)
                    if pair_counts[pair] is equal nil:
                        set pair_counts[pair] to 1
                    otherwise:
                        set pair_counts[pair] to pair_counts[pair] + 1
                    set i to i + 1

    // Find most frequent pairs and add as merges
    set merges to []
    set merge_idx to 0
    repeat while merge_idx is below num_merges:
        set best_pair to ""
        set best_count to 0
        repeat for each entry in pair_counts:
            if entry.value is above best_count:
                set best_count to entry.value
                set best_pair to entry.key
        if best_pair is not equal "":
            append best_pair to merges
            set pair_counts[best_pair] to 0
        set merge_idx to merge_idx + 1

    set bpe_merges to merges
    log "Learned " + string(len(merges)) + " BPE merges"
    respond with merges

// ── Encoding ───────────────────────────────────────────────

to encode with text:
    purpose: "Convert NC source text to token IDs"
    set tokens to [SPECIAL_TOKENS.BOS]

    // Try to match NC keywords first (greedy longest match)
    set i to 0
    repeat while i is below len(text):
        set matched to false

        // Try each keyword from longest to shortest
        set best_match to ""
        repeat for each kw in NC_KEYWORDS:
            set kw_len to len(kw)
            set window to substr(text, i, kw_len)
            if window is equal kw:
                if len(kw) is above len(best_match):
                    set best_match to kw

        if len(best_match) is above 0:
            set tok_id to vocab[best_match]
            if tok_id is not equal nil:
                append tok_id to tokens
            otherwise:
                append SPECIAL_TOKENS.UNK to tokens
            set i to i + len(best_match)
        otherwise:
            // Single character fallback
            set char to substr(text, i, 1)
            set tok_id to vocab[char]
            if tok_id is not equal nil:
                append tok_id to tokens
            otherwise:
                append SPECIAL_TOKENS.UNK to tokens
            set i to i + 1

    append SPECIAL_TOKENS.EOS to tokens
    respond with tokens

to encode_batch with texts:
    purpose: "Encode multiple texts with padding to same length"
    set all_tokens to []
    set max_len to 0

    // Encode all texts
    repeat for each text in texts:
        run encode with text
        set encoded to result
        append encoded to all_tokens
        if len(encoded) is above max_len:
            set max_len to len(encoded)

    // Pad to max length
    set padded to []
    repeat for each tokens in all_tokens:
        set padded_seq to tokens
        repeat while len(padded_seq) is below max_len:
            append SPECIAL_TOKENS.PAD to padded_seq
        append padded_seq to padded

    respond with {"tokens": padded, "max_len": max_len}

// ── Decoding ───────────────────────────────────────────────

to decode with token_ids:
    purpose: "Convert token IDs back to NC source text"
    set parts to []

    repeat for each tok_id in token_ids:
        // Skip special tokens
        if tok_id is not equal SPECIAL_TOKENS.PAD:
            if tok_id is not equal SPECIAL_TOKENS.BOS:
                if tok_id is not equal SPECIAL_TOKENS.EOS:
                    set token_str to reverse_vocab[tok_id]
                    if token_str is not equal nil:
                        append token_str to parts
                    otherwise:
                        append "<UNK>" to parts

    respond with join(parts, "")

to decode_batch with batch_token_ids:
    purpose: "Decode a batch of token ID sequences"
    set decoded_texts to []
    repeat for each token_ids in batch_token_ids:
        run decode with token_ids
        append result to decoded_texts
    respond with decoded_texts

// ── Save / Load ────────────────────────────────────────────

to save_vocab with path:
    purpose: "Save vocabulary to file"
    set vocab_data to {"vocab": vocab, "reverse_vocab": reverse_vocab, "bpe_merges": bpe_merges, "vocab_size": vocab_size}
    store vocab_data into path
    log "Vocabulary saved to " + path
    respond with true

to load_vocab with path:
    purpose: "Load vocabulary from file"
    gather vocab_data from path
    if vocab_data is not equal nil:
        set vocab to vocab_data.vocab
        set reverse_vocab to vocab_data.reverse_vocab
        set bpe_merges to vocab_data.bpe_merges
        set vocab_size to vocab_data.vocab_size
        log "Vocabulary loaded: " + string(vocab_size) + " tokens"
        respond with true
    otherwise:
        log "Failed to load vocab from " + path
        respond with false

// ── API ────────────────────────────────────────────────────

api:
    POST /encode        runs encode

// FILE: nc-ai-native/models/nc_transformer.nc

// ═══════════════════════════════════════════════════════════
//  NC Transformer Model — Written in NC Language
//
//  This is the NC AI brain written IN NC, not in C.
//  The NC engine (C) handles tensor math underneath,
//  just like PyTorch uses C++ under Python code.
//
//  Architecture:
//    Decoder-only transformer (GPT-style)
//    - Token embedding layer
//    - Positional encoding
//    - Multi-head self-attention blocks
//    - Feed-forward layers
//    - Layer normalization
//    - Language model head
//
//  Usage:
//    nc nc-ai-native/models/nc_transformer.nc
// ═══════════════════════════════════════════════════════════

service "nc-transformer"
version "1.0.0"

// ── Model Configuration ────────────────────────────────────

let model_config = {
    vocab_size:   4096,
    embed_dim:    256,
    num_heads:    8,
    num_layers:   6,
    ff_dim:       1024,
    max_seq_len:  512,
    dropout:      0.1
}

// ── Activation Functions ───────────────────────────────────

to relu with x:
    purpose: "Rectified Linear Unit activation"
    if x is above 0:
        respond with x
    otherwise:
        respond with 0

to gelu with x:
    purpose: "Gaussian Error Linear Unit — smoother than ReLU"
    // GELU(x) ≈ 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    set x3 to x * x * x
    set inner to 0.7978845608 * (x + 0.044715 * x3)
    set tanh_val to (exp(inner) - exp(0 - inner)) / (exp(inner) + exp(0 - inner))
    respond with 0.5 * x * (1 + tanh_val)

to softmax with scores:
    purpose: "Convert logits to probability distribution"
    set max_score to 0
    repeat for each s in scores:
        if s is above max_score:
            set max_score to s

    set exps to []
    set total to 0
    repeat for each s in scores:
        set e to exp(s - max_score)
        append e to exps
        set total to total + e

    set probs to []
    repeat for each e in exps:
        append e / total to probs

    respond with probs

to layer_norm with values and eps:
    purpose: "Normalize a layer's activations for stable training"
    set n to len(values)
    set mean to 0
    repeat for each v in values:
        set mean to mean + v
    set mean to mean / n

    set variance to 0
    repeat for each v in values:
        set diff to v - mean
        set variance to variance + diff * diff
    set variance to variance / n

    set normalized to []
    repeat for each v in values:
        set norm to (v - mean) / sqrt(variance + eps)
        append norm to normalized

    respond with normalized

// ── Attention Mechanism ────────────────────────────────────

to scaled_dot_product_attention with queries, keys, values, mask:
    purpose: "Core attention: how much each token attends to others"
    set d_k to model_config.embed_dim / model_config.num_heads
    set scale to sqrt(d_k)

    // Compute attention scores: Q * K^T / sqrt(d_k)
    set attention_scores to []
    repeat for each q in queries:
        set row_scores to []
        repeat for each k in keys:
            // Dot product of query and key vectors
            set dot to 0
            set i to 0
            repeat for each q_val in q:
                set dot to dot + q_val * get(k, i)
                set i to i + 1
            append dot / scale to row_scores
        append row_scores to attention_scores

    // Apply causal mask (prevent attending to future tokens)
    if mask is not equal nil:
        set masked_scores to []
        set row_i to 0
        repeat for each row in attention_scores:
            set masked_row to []
            set col_j to 0
            repeat for each score in row:
                set m to get(get(mask, row_i), col_j)
                if m is equal 0:
                    append score to masked_row
                otherwise:
                    append -1e9 to masked_row
                set col_j to col_j + 1
            append masked_row to masked_scores
            set row_i to row_i + 1
        set attention_scores to masked_scores

    // Softmax to get attention weights
    set attention_weights to []
    repeat for each row in attention_scores:
        run softmax with row
        append result to attention_weights

    // Weighted sum of values
    set output to []
    repeat for each weights in attention_weights:
        set weighted_sum to []
        repeat for each v in values:
            set idx to 0
            set weighted_v to []
            repeat for each val in v:
                append val * get(weights, idx) to weighted_v
                set idx to idx + 1
            append weighted_v to weighted_sum
        append weighted_sum to output

    respond with {"output": output, "weights": attention_weights}

to multi_head_attention with x, num_heads, embed_dim:
    purpose: "Split attention across multiple heads for richer representations"
    set head_dim to embed_dim / num_heads
    set all_head_outputs to []

    set head to 0
    repeat while head is below num_heads:
        // Each head gets a slice of the embedding
        set start to head * head_dim
        set q_slice to []
        set k_slice to []
        set v_slice to []

        repeat for each token in x:
            set q_part to []
            set k_part to []
            set v_part to []
            set idx to start
            repeat while idx is below start + head_dim:
                append get(token, idx) to q_part
                append get(token, idx) to k_part
                append get(token, idx) to v_part
                set idx to idx + 1
            append q_part to q_slice
            append k_part to k_slice
            append v_part to v_slice

        // Build causal mask
        set seq_len to len(x)
        set mask to []
        set i to 0
        repeat while i is below seq_len:
            set mask_row to []
            set j to 0
            repeat while j is below seq_len:
                if j is above i:
                    append 1 to mask_row
                otherwise:
                    append 0 to mask_row
                set j to j + 1
            append mask_row to mask
            set i to i + 1

        run scaled_dot_product_attention with q_slice, k_slice, v_slice, mask
        append result.output to all_head_outputs
        set head to head + 1

    respond with all_head_outputs

// ── Feed-Forward Network ───────────────────────────────────

to feed_forward with x, ff_dim:
    purpose: "Position-wise feed-forward transformation"
    // Two linear projections with GELU activation
    set hidden to []
    repeat for each token_vec in x:
        set h to []
        repeat while len(h) is below ff_dim:
            // Project to ff_dim (simplified — actual weights come from training)
            set idx to len(h) mod len(token_vec)
            run gelu with get(token_vec, idx) * 1.0
            append result to h
        append h to hidden

    // Project back to embed_dim
    set output to []
    repeat for each h_vec in hidden:
        set out_vec to []
        repeat while len(out_vec) is below model_config.embed_dim:
            set idx to len(out_vec) mod len(h_vec)
            append get(h_vec, idx) to out_vec
        append out_vec to output

    respond with output

// ── Transformer Block ──────────────────────────────────────

to transformer_block with x, layer_id:
    purpose: "Single transformer decoder block: attention + feed-forward"
    log "  Layer " + string(layer_id) + ": attention..."

    // Self-attention with residual connection
    run multi_head_attention with x, model_config.num_heads, model_config.embed_dim
    set attn_output to result

    // Merge attention heads back
    set merged to []
    repeat for each token_idx in range(len(x)):
        set token_out to []
        repeat for each head_out in attn_output:
            set head_token to get(head_out, token_idx)
            if head_token is not equal nil:
                repeat for each val in head_token:
                    append val to token_out
        append token_out to merged

    // Residual: x + attention(x)
    set residual1 to []
    set i to 0
    repeat for each original in x:
        set attn_tok to get(merged, i)
        set combined to []
        set j to 0
        repeat for each v in original:
            if attn_tok is not equal nil:
                set av to get(attn_tok, j)
                if av is not equal nil:
                    append v + av to combined
                otherwise:
                    append v to combined
            otherwise:
                append v to combined
            set j to j + 1
        append combined to residual1
        set i to i + 1

    // Layer norm after attention
    set normed1 to []
    repeat for each tok in residual1:
        run layer_norm with tok and 1e-5
        append result to normed1

    // Feed-forward with residual
    run feed_forward with normed1, model_config.ff_dim
    set ff_output to result

    set residual2 to []
    set i to 0
    repeat for each normed in normed1:
        set ff_tok to get(ff_output, i)
        set combined to []
        set j to 0
        repeat for each v in normed:
            if ff_tok is not equal nil:
                set fv to get(ff_tok, j)
                if fv is not equal nil:
                    append v + fv to combined
                otherwise:
                    append v to combined
            otherwise:
                append v to combined
            set j to j + 1
        append combined to residual2
        set i to i + 1

    // Final layer norm
    set output to []
    repeat for each tok in residual2:
        run layer_norm with tok and 1e-5
        append result to output

    respond with output

// ── Positional Encoding ────────────────────────────────────

to positional_encoding with seq_len, embed_dim:
    purpose: "Sinusoidal position encoding — gives tokens positional awareness"
    set encodings to []

    set pos to 0
    repeat while pos is below seq_len:
        set enc to []
        set i to 0
        repeat while i is below embed_dim:
            set freq to 1 / pow(10000, (2 * floor(i / 2)) / embed_dim)
            if i mod 2 is equal 0:
                append sin(pos * freq) to enc
            otherwise:
                append cos(pos * freq) to enc
            set i to i + 1
        append enc to encodings
        set pos to pos + 1

    respond with encodings

// ── Full Forward Pass ──────────────────────────────────────

to forward with token_ids:
    purpose: "Full transformer forward pass: tokens → logits"
    set seq_len to len(token_ids)
    log "Forward pass: seq_len=" + string(seq_len)

    // Token embeddings (lookup table — simplified)
    set embeddings to []
    repeat for each tok_id in token_ids:
        set emb to []
        set i to 0
        repeat while i is below model_config.embed_dim:
            // Deterministic embedding from token_id (real: learned weight matrix)
            set val to sin(tok_id * 0.01 + i * 0.1)
            append val to emb
            set i to i + 1
        append emb to embeddings

    // Add positional encoding
    run positional_encoding with seq_len, model_config.embed_dim
    set pos_enc to result

    set x to []
    set i to 0
    repeat for each emb in embeddings:
        set pos to get(pos_enc, i)
        set combined to []
        set j to 0
        repeat for each e_val in emb:
            set p_val to get(pos, j)
            if p_val is not equal nil:
                append e_val + p_val to combined
            otherwise:
                append e_val to combined
            set j to j + 1
        append combined to x
        set i to i + 1

    // Pass through transformer layers
    set layer to 0
    repeat while layer is below model_config.num_layers:
        run transformer_block with x, layer
        set x to result
        set layer to layer + 1

    // LM head: project to vocabulary logits
    set logits to []
    repeat for each tok in x:
        set vocab_scores to []
        set v to 0
        repeat while v is below model_config.vocab_size:
            // Simplified projection (real: learned weight matrix)
            set score to 0
            set i to 0
            repeat for each val in tok:
                set score to score + val * sin(v * 0.001 + i * 0.01)
                set i to i + 1
            append score to vocab_scores
            set v to v + 1
        append vocab_scores to logits

    respond with {"logits": logits, "hidden": x}

// ── API ────────────────────────────────────────────────────

api:
    POST /forward runs forward
    POST /attention runs scaled_dot_product_attention
    POST /encode runs positional_encoding

    purpose: "Split training pairs into mini-batches"

// ── Training Step ──────────────────────────────────────────

to train_step with batch, step:
    purpose: "Single training step on a mini-batch"
    set total_loss to 0
    set batch_size to len(batch)

    repeat for each pair in batch:
        // Encode input and target
        set input_text to pair.input
        set target_text to pair.output

        // Ask AI to generate NC code from input (teacher forcing)
        ask AI to "You are an NC code generator. Convert this request to NC code.\n\nRequest: {{input_text}}\n\nGenerate valid NC code:" save as generated

        // Compare generated vs target to compute loss signal
        if generated is equal target_text:
            set loss to 0
        otherwise:
            // Compute similarity-based loss
            set gen_len to len(generated)
            set tgt_len to len(target_text)
            set match_count to 0
            set i to 0
            repeat while i is below min(gen_len, tgt_len):
                if substr(generated, i, 1) is equal substr(target_text, i, 1):
                    set match_count to match_count + 1
                set i to i + 1
            set similarity to match_count / max(gen_len, tgt_len)
            set loss to 1 - similarity

        set total_loss to total_loss + loss

    set avg_loss to total_loss / batch_size
    respond with {"loss": avg_loss, "step": step}

to evaluate with eval_pairs:
    purpose: "Evaluate model quality on held-out examples"
    log "Evaluating on " + string(len(eval_pairs)) + " examples..."

    set total_loss to 0
    set exact_matches to 0

    repeat for each pair in eval_pairs:
        ask AI to "Convert this request to NC code.\n\nRequest: {{pair.input}}\n\nGenerate NC code:" save as generated

        // Check exact match
        if generated is equal pair.output:
            set exact_matches to exact_matches + 1

        // Compute loss
        set gen_len to len(generated)
        set tgt_len to len(pair.output)
        set match_count to 0
        set i to 0
        repeat while i is below min(gen_len, tgt_len):
            if substr(generated, i, 1) is equal substr(pair.output, i, 1):
                set match_count to match_count + 1
            set i to i + 1
        set similarity to match_count / max(gen_len, tgt_len)
        set total_loss to total_loss + (1 - similarity)

    set avg_loss to total_loss / len(eval_pairs)
    set accuracy to exact_matches / len(eval_pairs)

    log "  Eval loss: " + string(avg_loss)
    log "  Exact match: " + string(exact_matches) + "/" + string(len(eval_pairs))

    respond with {
        "eval_loss":     avg_loss,
        "exact_matches": exact_matches,
        "accuracy":      accuracy,
        "total":         len(eval_pairs)
    }

to train:
    purpose: "Full training loop — trains NC AI to generate NC code"
    log "══════════════════════════════════════════"

    run load_training_data with train_config.data_path
    set all_pairs to result

    // Split train/eval (90/10)
    set split_idx to floor(len(all_pairs) * 0.9)
    set train_pairs to []
    set eval_pairs to []
    set i to 0
    repeat for each pair in all_pairs:

            append pair to train_pairs

            append pair to eval_pairs
        set i to i + 1

    log "Eval examples:  " + string(len(eval_pairs))
    log ""

    set global_step to 0
    set best_eval_loss to 999999

    // Training epochs
    set epoch to 0
    repeat while epoch is below train_config.epochs:

            set global_step to global_step + 1

            if global_step mod train_config.log_every is equal 0:

                run evaluate with eval_pairs
                set eval_result to result

                    set best_eval_loss to eval_result.eval_loss
                    log "  * New best model! Saving..."
                    store {"epoch": epoch, "step": global_step, "eval_loss": best_eval_loss} into train_config.checkpoint_dir + "/best_model.json"

            if global_step mod train_config.save_every is equal 0:

    log "Best eval loss: " + string(best_eval_loss)
    log "Checkpoints saved to: " + train_config.checkpoint_dir

        "total_steps":    global_step,
