// ═══════════════════════════════════════════════════════════
//  NeuralEdge Stock Service — Written in NC Language
//
//  Full stock market intelligence backend — AI-powered
//  technical analysis, price predictions, portfolio management.
//
//  All in NC. No Python. No Node. No Java. Just NC.

//  Start: nc nc-apps/stock-market/backend/stock_service.nc
//  Port:  8000

service "neuraledge-stock"
version "1.0.0"

configure:
    ai_model is "default"
    port is 8000
    ai_system_prompt is "You are NeuralEdge AI, a financial analysis assistant. Provide accurate, data-driven analysis of stocks. Always include specific numbers and technical indicators. Format responses as JSON when requested."

let config = {
    provider:       env("MARKET_DATA_PROVIDER", "yahoo"),
    api_key:        env("MARKET_DATA_API_KEY", ""),
    cache_ttl:      300,
    horizon_days:   7,
    model:          env("OLLAMA_MODEL", "llama3"),
    ollama_host:    env("OLLAMA_HOST", "http://localhost:11434")
}

// ── Health ─────────────────────────────────────────────────

to health:
    respond with {
        "status":    "ok",
        "service":   "NeuralEdge",
        "version":   "1.0.0",
        "ai_model":  config.model,
        "timestamp": now()
    }

// ── Market Data ────────────────────────────────────────────

to get_quote with symbol:
    purpose: "Get current stock price and day data"
    set sym to upper(symbol)

    // In production: fetch from config.provider API
    // For demo: AI generates realistic quote data
    ask AI to "Generate realistic current stock data for {{sym}}. Return JSON: {symbol, price (number), change (number), change_pct (number), volume (number), market_cap (string), pe_ratio (number), week_52_high (number), week_52_low (number)}" save as quote_data

    respond with quote_data

to get_history with symbol, period:
    purpose: "Get historical OHLCV price data"
    set sym to upper(symbol)
    set days to 30
    if period is equal "1w":
        set days to 7
    if period is equal "3m":
        set days to 90
    if period is equal "1y":
        set days to 365

    ask AI to "Generate realistic historical OHLCV data for {{sym}} for the last {{days}} days. Return JSON array of {date, open, high, low, close, volume} objects. Use realistic price ranges." save as history

    respond with {"symbol": sym, "period": period, "data": history}

to get_sector_performance:
    purpose: "Get performance by market sector"
    ask AI to "Return current US stock sector performance. JSON: [{sector, performance_pct, top_stock, description}] for: Technology, Healthcare, Finance, Energy, Consumer, Industrial, Materials, Utilities, Real Estate, Communication" save as sectors

    respond with sectors

// ── Technical Analysis ─────────────────────────────────────

to analyze with symbol:
    purpose: "Full technical analysis: RSI, MACD, Bollinger, moving averages"
    set sym to upper(symbol)

    ask AI to "Perform comprehensive technical analysis for {{sym}}. Return JSON: {symbol, rsi_14 (number 0-100), macd_signal (bullish/bearish/neutral), macd_value (number), macd_histogram (number), sma_20 (number), sma_50 (number), sma_200 (number), bollinger_upper (number), bollinger_lower (number), bollinger_mid (number), volume_sma_20 (number), atr_14 (number), support_level (number), resistance_level (number), trend (uptrend/downtrend/sideways), overall_signal (strong_buy/buy/hold/sell/strong_sell), confidence (0-100)}" save as analysis

    respond with analysis

to rsi with symbol, period:
    purpose: "Calculate RSI for a stock"
    set sym to upper(symbol)
    set p to 14
    if period is not equal nil:
        set p to period

    ask AI to "Calculate RSI({{p}}) for {{sym}}. Return JSON: {symbol, rsi, interpretation (overbought/neutral/oversold), signal (sell/hold/buy)}" save as rsi_data
    respond with rsi_data

to macd with symbol:
    purpose: "MACD signal and histogram"
    set sym to upper(symbol)
    ask AI to "Calculate MACD(12,26,9) for {{sym}}. Return JSON: {symbol, macd_line, signal_line, histogram, crossover (bullish/bearish/none), trend_strength (strong/moderate/weak)}" save as macd_data
    respond with macd_data

to bollinger with symbol:
    purpose: "Bollinger Bands analysis"
    set sym to upper(symbol)
    ask AI to "Calculate Bollinger Bands(20,2) for {{sym}}. Return JSON: {symbol, upper_band, middle_band, lower_band, current_price, bandwidth_pct, position (above_upper/near_upper/middle/near_lower/below_lower), squeeze (boolean)}" save as bb_data
    respond with bb_data

// ── AI Predictions ─────────────────────────────────────────

to predict with symbol:
    purpose: "AI price prediction for next 7 days"
    set sym to upper(symbol)

    // Get technical context first
    run analyze with sym
    set ta to result

    ask AI to "Based on this technical analysis for {{sym}}, predict the price for the next {{config.horizon_days}} days. Context: {{ta}}\n\nReturn JSON: {symbol, current_price, predictions: [{day, date, predicted_price, low_estimate, high_estimate, confidence}], overall_direction (bullish/bearish/neutral), key_levels: {support, resistance, target_1, target_2, stop_loss}, risk_factors: [string], catalysts: [string], recommendation (strong_buy/buy/hold/sell/strong_sell), position_size_suggestion (string)}" save as prediction

    respond with prediction

to predict_batch with symbols:
    purpose: "Batch predictions for multiple symbols"
    set results to []
    repeat for each sym in symbols:
        run predict with sym
        append result to results
    respond with results

// ── Sentiment Analysis ─────────────────────────────────────

to sentiment with symbol:
    purpose: "AI news and social sentiment analysis"
    set sym to upper(symbol)

    ask AI to "Analyze market sentiment for {{sym}} based on recent news, analyst reports, and social media. Return JSON: {symbol, overall_sentiment (very_bullish/bullish/neutral/bearish/very_bearish), sentiment_score (-100 to 100), news_sentiment (string), analyst_consensus (string), retail_sentiment (string), institutional_activity (string), recent_news: [{headline, sentiment, impact}], risk_events: [string]}" save as sentiment_data

    respond with sentiment_data

// ── Portfolio Management ───────────────────────────────────

to analyze_portfolio with holdings:
    purpose: "Analyze portfolio risk, diversification, and recommendations"
    ask AI to "Analyze this stock portfolio and provide recommendations. Holdings: {{holdings}}\n\nReturn JSON: {total_value, day_pnl, total_return_pct, beta, sharpe_ratio, diversification_score (0-100), risk_level (low/moderate/high/very_high), sector_breakdown: {}, concentration_risk: [string], recommendations: [string], rebalancing_suggestions: [{action, symbol, reason}]}" save as portfolio_analysis

    respond with portfolio_analysis

to suggest_rebalance with holdings, target_allocation:
    purpose: "Suggest portfolio rebalancing to hit target allocation"
    ask AI to "Suggest rebalancing trades to move from current holdings to target allocation.\n\nCurrent: {{holdings}}\nTarget: {{target_allocation}}\n\nReturn JSON: [{action (buy/sell), symbol, quantity, estimated_value, reason}]" save as rebalance

    respond with rebalance

// ── Screener ───────────────────────────────────────────────

to screen with criteria:
    purpose: "Screen stocks matching criteria"
    ask AI to "Find stocks matching these criteria: {{criteria}}. Return JSON: [{symbol, name, price, match_score (0-100), why_matches: [string]}] — top 10 matches." save as matches
    respond with matches

// ── Watchlist ──────────────────────────────────────────────

to watchlist_summary with symbols:
    purpose: "Quick summary of a watchlist"
    set summaries to []
    repeat for each sym in symbols:
        run get_quote with sym
        append result to summaries
    respond with summaries

// ── Lifecycle ──────────────────────────────────────────────

on start {
    log("NeuralEdge AI Stock Service starting on port " + string(config.port))
    log("Provider: " + config.provider)
    log("AI Model: " + config.model)
    log("Powered by NC Language — DevHeal Labs AI")
}

// ── API ────────────────────────────────────────────────────

api:
    GET  /health                    runs health
    GET  /api/v1/quote/:symbol      runs get_quote
    GET  /api/v1/history/:symbol    runs get_history
    GET  /api/v1/sectors            runs get_sector_performance
    POST /api/v1/analyze/:symbol    runs analyze
    GET  /api/v1/rsi/:symbol        runs rsi
    GET  /api/v1/macd/:symbol       runs macd
    GET  /api/v1/bollinger/:symbol  runs bollinger
    POST /api/v1/predict/:symbol    runs predict
    POST /api/v1/predict/batch      runs predict_batch
    POST /api/v1/sentiment/:symbol  runs sentiment
    POST /api/v1/portfolio/analyze  runs analyze_portfolio
    POST /api/v1/portfolio/rebalance runs suggest_rebalance
    POST /api/v1/screen             runs screen
    POST /api/v1/watchlist          runs watchlist_summary
