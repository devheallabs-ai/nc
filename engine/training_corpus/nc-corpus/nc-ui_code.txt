// ═══════════════════════════════════════════════════════════
//  NC UI CLI — Written in NC Language
//
//  Replaces cli.js — the NC UI command-line tool.
//  Written in NC, not Node.js/JavaScript.
//
//  Usage:
//    nc nc-ui/cli.nc build   app.ncui
//    nc nc-ui/cli.nc serve   app.ncui --port 3000
//    nc nc-ui/cli.nc watch   app.ncui
//    nc nc-ui/cli.nc check   app.ncui
//    nc nc-ui/cli.nc tokens  app.ncui
//    nc nc-ui/cli.nc ast     app.ncui
//    nc nc-ui/cli.nc new     myapp
//    nc nc-ui/cli.nc help
//
//  Calls the NC UI Compiler service (compiler.nc)
//  to do the actual compilation.
// ═══════════════════════════════════════════════════════════

service "nc-ui-cli"
version "3.0.0"

import "subprocess"
import "os"

let compiler_port to 7600
let compiler_url to "http://localhost:" + string(compiler_port)
let emitter_port to 7605
let emitter_url to "http://localhost:" + string(emitter_port)

// ── Compiler Client ────────────────────────────────────────

to compile_source with source:
    purpose: "Send source to NC UI compiler and get HTML back"
    set url to compiler_url + "/compile"
    set payload to {"source": source}
    fetch url method "POST" body payload save as result
    if result is equal nil:
        log "ERROR: NC UI compiler not running."
        log "Start it with: nc serve nc-ui/compiler.nc"
        respond with {"error": "Compiler not running"}
    respond with result

to compile_file_path with file_path:
    purpose: "Read and compile a .ncui file"
    gather source from file_path
    if source is equal nil:
        log "ERROR: File not found: " + file_path
        respond with {"error": "File not found"}
    log "Compiling: " + file_path + " ..."
    run compile_source with source
    respond with result

// ── Build Command ──────────────────────────────────────────

to cmd_build with file_path, output_path:
    purpose: "Compile .ncui file to HTML"
    run compile_file_path with file_path
    set result_data to result

    if result_data.error is not equal nil:
        log "Build failed: " + result_data.error
        respond with {"success": false}

    set html to result_data.html

    // Determine output path
    set out_path to output_path
    if out_path is equal nil:
        set out_path to replace(file_path, ".ncui", ".html")

    store html into out_path
    log "Built: " + file_path + " → " + out_path
    log "Size: " + string(len(html)) + " bytes"
    respond with {"success": true, "output": out_path, "size": len(html)}

// ── Serve Command ──────────────────────────────────────────

to cmd_serve with file_path, port:
    purpose: "Compile and serve .ncui file on HTTP"
    set serve_port to port
    if serve_port is equal nil:
        set serve_port to 3000

    // Initial build
    run compile_file_path with file_path
    if result.error is not equal nil:
        log "ERROR: " + result.error
        respond with {"success": false}

    set html to result.html
    store html into "./.ncui_serve_temp.html"

    log ""
    log "  NC UI Dev Server"
    log "  ══════════════════════════════════"
    log "  File:   " + file_path
    log "  URL:    http://localhost:" + string(serve_port)
    log "  Press Ctrl+C to stop"
    log ""

    // Serve HTML on the port
    set serve_service to "service \"ncui-dev\"\nconfigure:\n    port is " + string(serve_port) + "\n\nto serve:\n    gather content from \"./.ncui_serve_temp.html\"\n    respond with content\n\napi:\n    GET / runs serve\n    GET /* runs serve"

    store serve_service into "./.ncui_dev_server.nc"
    run subprocess.run with "nc serve ./.ncui_dev_server.nc"

    respond with {"success": true, "port": serve_port}

// ── Watch Command ──────────────────────────────────────────

to cmd_watch with file_path, port:
    purpose: "Watch .ncui file and rebuild on change"
    set watch_port to port
    if watch_port is equal nil:
        set watch_port to 3000

    log ""
    log "  NC UI Watch Mode"
    log "  ══════════════════════════════════"
    log "  File:   " + file_path
    log "  URL:    http://localhost:" + string(watch_port)
    log "  Press Ctrl+C to stop"
    log ""

    // Initial build
    run cmd_build with file_path, nil

    // Watch loop
    set last_mtime to os.mtime(file_path)
    repeat while true:
        set current_mtime to os.mtime(file_path)
        if current_mtime is above last_mtime:
            log "Change detected — rebuilding..."
            run cmd_build with file_path, nil
            set last_mtime to current_mtime
        wait 500

    respond with {"success": true}

// ── Check / Validate Command ───────────────────────────────

to cmd_check with file_path:
    purpose: "Validate .ncui syntax without compiling"
    gather source from file_path
    if source is equal nil:
        log "ERROR: File not found: " + file_path
        respond with {"valid": false}

    set url to compiler_url + "/validate"
    fetch url method "POST" body {"source": source} save as result

    if result is equal nil:
        log "ERROR: NC UI compiler not running."
        respond with {"valid": false}

    if result.valid:
        log file_path + ": OK — " + string(result.token_count) + " tokens"
    otherwise:
        log file_path + ": ERRORS"
        repeat for each err in result.errors:
            log "  - " + err

    respond with result

// ── Tokens Command ─────────────────────────────────────────

to cmd_tokens with file_path:
    purpose: "Show lexer tokens for debugging"
    gather source from file_path
    if source is equal nil:
        log "ERROR: File not found: " + file_path
        respond with {"error": "File not found"}

    set url to compiler_url + "/tokens"
    fetch url method "POST" body {"source": source} save as result

    log "Tokens for: " + file_path
    log "═══════════════════════════"
    repeat for each tok in result.tokens:
        log tok.type + " '" + tok.value + "' (line " + string(tok.line) + ")"
    log ""
    log "Total: " + string(result.count) + " tokens"
    respond with result

// ═══════════════════════════════════════════════════════════
//  NC UI Compiler — Written in NC Language
//
//  This replaces compiler.js.
//  The NC UI compiler is now NC code, not JavaScript.
//
//  Why JS was there before:
//    - NC UI was built before NC was mature
//    - JS runs in browsers AND Node.js (universal)
//    - But it breaks the NC-first principle
//
//  This NC compiler:
//    - Runs as an NC service (nc serve compiler.nc)
//    - Compiles .ncui files to HTML+CSS+JS via HTTP
//    - Handles ALL enterprise NC UI v2.0 syntax
//    - Browser receives compiled HTML — no NC engine in browser
//
//  Enterprise NC UI v2.0 Features:
//    - auth: bearer, session, OAuth, SAML
//    - guard: route guards with role checking
//    - routes: full routing with visibility
//    - state: global reactive state with types
//    - theme: full CSS variable system
//    - component: lifecycle, props, computed, slots
//    - page: lifecycle, guards, state binding
//    - form: declarative validation
//    - computed: reactive computed properties
//    - on mount / on unmount: lifecycle hooks
//    - fetch with auth: HTTP + auth headers built-in
//    - emit: event bus
//    - watch: reactive watchers
//    - i18n: built-in localization
//    - rbac: role-based access control
//    - error-boundary: graceful error handling
//    - skeleton: loading states
//    - middleware: request/response interceptors
//
//  Start: nc serve nc-ui/compiler.nc
//  Port:  7600
//
//  Compile via API:
//    POST /compile         { source: "..." }  → { html: "..." }
//    POST /compile/file    { path: "app.ncui" } → { html: "..." }
//    POST /ast             { source: "..." }  → { ast: {...} }
//    POST /tokens          { source: "..." }  → { tokens: [...] }
//    POST /validate        { source: "..." }  → { valid: bool, errors: [...] }
// ═══════════════════════════════════════════════════════════

service "nc-ui-compiler"
version "2.0.0"

configure:
    port is 7600
    ai_model is "default"

to lex with source:
    purpose: "Tokenize NC UI v2 source into a flat list of tokens"
    set tokens to []
    set i to 0
    set line to 1
    set col to 1
    set indent_stack to [0]

    repeat while i is below len(source):
        set ch to substr(source, i, 1)

        // Newline
        if ch is equal "\n":
            append {"type": "NEWLINE", "value": "\n", "line": line, "col": col} to tokens
            set line to line + 1
            set col to 1
            set i to i + 1

            // Count indent on next line
            set spaces to 0
            repeat while i + spaces is below len(source):
                set nc to substr(source, i + spaces, 1)
                if nc is equal " ":
                    set spaces to spaces + 1
                otherwise:
                    if nc is equal "\t":
                        set spaces to spaces + 4
                    otherwise:
                        // Stop
                        set nc to "done"
                if nc is equal "done":
                    // Exit loop
                    set spaces to spaces
                    set nc to ""

            set current_indent to get(indent_stack, len(indent_stack) - 1)
            if spaces is above current_indent:
                append spaces to indent_stack
                append {"type": "INDENT", "value": spaces, "line": line, "col": 1} to tokens
                set i to i + spaces
                set col to spaces + 1
            otherwise:
                repeat while spaces is below current_indent:
                    remove from indent_stack at len(indent_stack) - 1
                    set current_indent to get(indent_stack, len(indent_stack) - 1)
                    append {"type": "DEDENT", "value": spaces, "line": line, "col": 1} to tokens
                set i to i + spaces
                set col to spaces + 1

        // Skip spaces (mid-line)
        otherwise:
            if ch is equal " " or ch is equal "\t":
                set i to i + 1
                set col to col + 1

            // Comment
            otherwise:
                if ch is equal "/" and substr(source, i, 2) is equal "//":
                    repeat while i is below len(source) and substr(source, i, 1) is not equal "\n":
                        set i to i + 1

                // String literal
                otherwise:
                    if ch is equal "\"":
                        set j to i + 1
                        set str to ""
                        repeat while j is below len(source):
                            set sc to substr(source, j, 1)
                            if sc is equal "\"":
                                set j to j + 1
                                // Exit
                                set sc to "done"
                            otherwise:
                                if sc is equal "done":
                                    set sc to ""
                                otherwise:
                                    set str to str + sc
                                    set j to j + 1
                        append {"type": "STRING", "value": str, "line": line, "col": col} to tokens
                        set col to col + (j - i)
                        set i to j

                    // Number
                    otherwise:
                        set digit to false
                        if ch is equal "0" or ch is equal "1" or ch is equal "2" or ch is equal "3" or ch is equal "4":
                            set digit to true
                        if ch is equal "5" or ch is equal "6" or ch is equal "7" or ch is equal "8" or ch is equal "9":
                            set digit to true
                        if digit:
                            set j to i
                            set num to ""
                            set has_dot to false
                            repeat while j is below len(source):
                                set nc2 to substr(source, j, 1)
                                if nc2 is equal ".":
                                    set has_dot to true
                                    set num to num + nc2
                                    set j to j + 1
                                otherwise:
                                    set is_num to false
                                    if nc2 is equal "0" or nc2 is equal "1" or nc2 is equal "2" or nc2 is equal "3" or nc2 is equal "4":
                                        set is_num to true
                                    if nc2 is equal "5" or nc2 is equal "6" or nc2 is equal "7" or nc2 is equal "8" or nc2 is equal "9":
                                        set is_num to true
                                    if is_num:
                                        set num to num + nc2
                                        set j to j + 1
                                    otherwise:
                                        set nc2 to "done"
                            if contains(get(tokens, len(tokens) - 1), "PERCENT") or substr(source, j, 1) is equal "%":
                                append {"type": "PERCENT", "value": num, "line": line, "col": col} to tokens
                                set j to j + 1
                            otherwise:
                                append {"type": "NUMBER", "value": num, "line": line, "col": col} to tokens
                            set col to col + (j - i)
                            set i to j

                        // Punctuation
                        otherwise:
                            if ch is equal ":" or ch is equal "," or ch is equal "[" or ch is equal "]" or ch is equal "{" or ch is equal "}":
                                append {"type": "PUNCT", "value": ch, "line": line, "col": col} to tokens
                                set i to i + 1
                                set col to col + 1

                            // Identifier or keyword
                            otherwise:
                                set is_alpha to false
                                set alpha_chars to "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_-"
                                set ci to 0
                                repeat while ci is below len(alpha_chars):
                                    if ch is equal substr(alpha_chars, ci, 1):
                                        set is_alpha to true
                                    set ci to ci + 1

                                if is_alpha:
                                    set j to i
                                    set word to ""
                                    repeat while j is below len(source):
                                        set wc to substr(source, j, 1)
                                        set wc_ok to false
                                        set wci to 0
                                        set valid_word_chars to "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_-"
                                        repeat while wci is below len(valid_word_chars):
                                            if wc is equal substr(valid_word_chars, wci, 1):
                                                set wc_ok to true
                                            set wci to wci + 1
                                        if wc_ok:
                                            set word to word + wc
                                            set j to j + 1
                                        otherwise:
                                            set wc to "done"

                                    set is_kw to false
                                    repeat for each kw in KEYWORDS:
                                        if lower(word) is equal kw:
                                            set is_kw to true
                                    if is_kw:
                                        append {"type": "KEYWORD", "value": lower(word), "line": line, "col": col} to tokens
                                    otherwise:
                                        append {"type": "IDENT", "value": word, "line": line, "col": col} to tokens
                                    set col to col + (j - i)
                                    set i to j
                                otherwise:
                                    // Unknown char — skip
                                    set i to i + 1
                                    set col to col + 1

    // Final DEDENT tokens to close all open blocks
    repeat while len(indent_stack) is above 1:
        remove from indent_stack at len(indent_stack) - 1
        append {"type": "DEDENT", "value": 0, "line": line, "col": 1} to tokens

    append {"type": "EOF", "value": "", "line": line, "col": col} to tokens
    respond with tokens

    purpose: "Generate enterprise security JavaScript for browser"
    respond with "// NC UI Enterprise Security Module\n(function(window) {\n'use strict';\nconst NCUISecurity = {\n  xss: {\n    sanitize(s) {\n      if (typeof s !== 'string') return '';\n      return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/\"/g,'&quot;').replace(/'/g,'&#x27;').replace(/`/g,'&#x60;').replace(/\\//g,'&#x2F;');\n    },\n    sanitizeUrl(url) {\n      if (!url || typeof url !== 'string') return '#';\n      const u = url.trim().toLowerCase();\n      if (u.startsWith('javascript:') || u.startsWith('data:') || u.startsWith('vbscript:')) return '#';\n      return url;\n    }\n  },\n  csrf: {\n    _tokens: new Set(),\n    generate() {\n      const b = new Uint8Array(32);\n      crypto.getRandomValues(b);\n      const t = btoa(String.fromCharCode(...b)).replace(/\\+/g,'-').replace(/\\//g,'_').replace(/=/g,'');\n      this._tokens.add(t);\n      setTimeout(() => this._tokens.delete(t), 3600000);\n      return t;\n    },\n    validate(t) {\n      if (!this._tokens.has(t)) return false;\n      this._tokens.delete(t);\n      return true;\n    }\n  },\n  jwt: {\n    _store: null,\n    decode(t) {\n      try {\n        const [,p] = t.split('.');\n        return JSON.parse(atob(p.replace(/-/g,'+').replace(/_/g,'/')));\n      } catch { return null; }\n    },\n    isExpired(t) {\n      const p = this.decode(t);\n      if (!p || !p.exp) return true;\n      return Date.now() / 1000 > p.exp;\n    },\n    store(t) { this._store = t; try { sessionStorage.setItem('ncui_jwt', t); } catch {} },\n    retrieve() { return this._store || ((() => { try { return sessionStorage.getItem('ncui_jwt'); } catch { return null; } })()); },\n    clear() { this._store = null; try { sessionStorage.removeItem('ncui_jwt'); } catch {} },\n    isAuthenticated() { const t = this.retrieve(); return !!t && !this.isExpired(t); },\n    getClaims() { const t = this.retrieve(); return t ? to define_policy with roles:
    purpose: "Define RBAC policy from NC UI roles config"
    set policy to {}
    repeat for each role in roles:
        set policy[role.name] to role.permissions
    respond with policy

to check_permission with policy, user_roles, action, resource:
    purpose: "Check if user has permission to perform action on resource"
    repeat for each role in user_roles:
        set perms to policy[role]
        if perms is not equal nil:
            repeat for each perm in perms:
                if perm is equal "*":
                    respond with {"allowed": true, "reason": "Role " + role + " has wildcard permission"}
                if perm is equal action + ":*":
                    respond with {"allowed": true, "reason": "Role " + role + " has " + action + " on all resources"}
                if perm is equal action + ":" + resource:
                    respond with {"allowed": true, "reason": "Role " + role + " has " + action + " on " + resource}
                if perm is equal "*:" + resource:
                    respond with {"allowed": true, "reason": "Role " + role + " has all actions on " + resource}

    respond with {"allowed": false, "reason": "No matching permission for " + action + ":" + resource}

// ── GitHub Pages deploy helper ────────────────────────────────

to prepare_github_pages with ncui_file_path, out_dir:
    purpose: "Prepare all files needed for GitHub Pages deployment"

    set dir to out_dir
    if dir is equal nil:
        set dir to "./dist"

    // Compile .ncui → .ncui.js
    run emit_js with ncui_file_path
    set js_result to result

    if js_result.error is not equal nil:
        log "ERROR: " + js_result.error
        respond with {"success": false}

    // Copy runtime
    gather runtime from "nc-ui/runtime/ncui-runtime.js"
    store runtime into dir + "/ncui-runtime.js"

    // Copy compiled app JS
    set js_filename to replace(ncui_file_path, ".ncui", ".ncui.js")
    gather js_content from js_filename
    store js_content into dir + "/" + js_filename

    // Generate index.html
    set app_name to "NC UI App"
    run emit_html_template with app_name, js_filename
    store result into dir + "/index.html"

    log ""
    log "  GitHub Pages Ready!"
    log "  ═════════════════════════════════════════"
    log "  Output folder: " + dir
    log "  Files:"
    log "    " + dir + "/index.html"
    log "    " + dir + "/ncui-runtime.js"
    log "    " + dir + "/app.ncui.js"
    log ""
    log "  Deploy:"
    log "    git add " + dir
    log "    git commit -m 'Deploy NC UI app'"
    log "    git push"
    log "    → Enable GitHub Pages in repo Settings → Pages → /" + dir
    log ""

    respond with {"success": true, "dir": dir}
