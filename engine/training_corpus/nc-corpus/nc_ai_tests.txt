    show "  [4/5] Distill & Learn Tests"
    set r4 to run_test_suite("nc-ai/tests/test_ai_distill.nc")
    set total_passed to total_passed + r4.passed
    set total_failed to total_failed + r4.failed

    // Test Suite 5: Package Export
    show "  [5/5] Export Tests"
    set r5 to run_test_suite("nc-ai/tests/test_ai_export.nc")

    set total_failed to total_failed + r5.failed

    show "  ═══════════════════════════"

to test_distill_prompts_exist:
    purpose: "Should have distillation prompts"
    set prompts to get_distill_prompts()
    assert length(prompts) >= 20, "Should have at least 20 prompts"
    respond with "ok"

to test_distill_prompts_diverse:
    purpose: "Prompts should cover different domains"
    set prompts to get_distill_prompts()
    set domains to {api: false, auth: false, dashboard: false, chat: false}

    repeat for each p in prompts:
        if p contains "api" or p contains "rest":
            set domains.api to true
        if p contains "auth" or p contains "login":
            set domains.auth to true
        if p contains "dashboard" or p contains "metrics":
            set domains.dashboard to true
        if p contains "chat" or p contains "message":
            set domains.chat to true

    assert domains.api is true, "Should have API prompts"
    assert domains.auth is true, "Should have auth prompts"
    assert domains.dashboard is true, "Should have dashboard prompts"
    assert domains.chat is true, "Should have chat prompts"
    respond with "ok"

to test_validate_nc_code_valid:
    purpose: "Valid NC code should pass validation"

    assert valid is true, "Valid NC code should pass"
    respond with "ok"
