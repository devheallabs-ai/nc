to detect_intent with prompt:
    purpose: "Classify user intent from natural language"
    set p to lower(prompt)
    set intent to "service"
    set features to []

    // Detect service type
    if contains(p, "crud") or contains(p, "database") or contains(p, "store"):
        set intent to "crud"
    if contains(p, "chat") or contains(p, "conversation") or contains(p, "talk"):
        set intent to "chatbot"
    if contains(p, "classify") or contains(p, "categorize") or contains(p, "label"):
        set intent to "classifier"
    if contains(p, "summarize") or contains(p, "summary") or contains(p, "tldr"):
        set intent to "summarizer"
    if contains(p, "pipeline") or contains(p, "process") or contains(p, "transform"):
        set intent to "pipeline"
    if contains(p, "webhook") or contains(p, "hook") or contains(p, "event"):
        set intent to "webhook"
    if contains(p, "dashboard") or contains(p, "analytics") or contains(p, "metrics"):
        set intent to "dashboard"
    if contains(p, "ui") or contains(p, "page") or contains(p, "frontend"):
        set intent to "ncui"

    // Detect features
    if contains(p, "auth") or contains(p, "login") or contains(p, "jwt"):
        append "auth" to features
    if contains(p, "ai") or contains(p, "gpt") or contains(p, "llm") or contains(p, "intelligent"):
        append "ai" to features
    if contains(p, "rate limit") or contains(p, "throttle"):
        append "rate_limit" to features
    if contains(p, "cache") or contains(p, "redis"):
        append "cache" to features
    if contains(p, "search") or contains(p, "find") or contains(p, "query"):
        append "search" to features
    if contains(p, "upload") or contains(p, "file"):
        append "upload" to features
    if contains(p, "email") or contains(p, "notify") or contains(p, "notification"):
        append "notification" to features
    if contains(p, "test") or contains(p, "testing"):
        append "test" to features
    if contains(p, "real-time") or contains(p, "websocket") or contains(p, "live"):
        append "realtime" to features

    // Extract service name from prompt
    set name to "my-service"
    set words to split(p, " ")
    repeat for each word in words:
        if len(word) is above 3:
            if not contains(word, "create"):
                if not contains(word, "build"):
                    if not contains(word, "make"):
                        if not contains(word, "service"):
                            if not contains(word, "with"):
                                set name to word
                                // Use first meaningful word

    respond with {
        "intent": intent,
        "features": features,
        "name": name,
        "raw_prompt": prompt
    }

// ── Code Validation ──────────────────────────────────────────

to validate_code with code:
    purpose: "Validate generated NC code"
    set issues to []

    if not contains(code, "service") and not contains(code, "page"):
        append "Missing service/page declaration" to issues

    if not contains(code, "to ") and not contains(code, "section"):
        append "No function definitions found" to issues

    // Check balanced braces
    set opens to 0
    set closes to 0
    set i to 0
    repeat while i is below len(code):
        set ch to substr(code, i, 1)
        if ch is equal "{":
            set opens to opens + 1
        if ch is equal "}":
            set closes to closes + 1
        set i to i + 1

    if opens is not equal closes:
        append "Unbalanced braces" to issues

    respond with {
        "valid": len(issues) is equal 0,
        "issues": issues
    }

// ═══════════════════════════════════════════════════════════
//  NC BPE Tokenizer — Written 100% in NC Language
//
//  Byte-Pair Encoding tokenizer trained on NC source code.
//  Converts NC text ↔ token IDs for the transformer.
//
//  Usage:
//    nc nc-ai/nc/models/tokenizer.nc
// ═══════════════════════════════════════════════════════════

service "nc-tokenizer"
version "2.0.0"

// ── Build Vocabulary ────────────────────────────────────────

to build_vocab with corpus_texts:
    purpose: "Build vocabulary from NC source texts"
    log "Building NC tokenizer vocabulary..."

    // Special tokens + NC keywords
    set vocab to {"<PAD>": 0, "<BOS>": 1, "<EOS>": 2, "<UNK>": 3}
    set vocab_size to 4

    // Add all NC keywords
    set keywords to [
        "service", "version", "configure", "to", "with", "purpose",
        "set", "ask", "AI", "save", "as", "respond",
        "if", "is", "above", "below", "equal", "not", "otherwise",
        "repeat", "for", "each", "in", "while", "times",
        "append", "run", "result", "define",
        "api", "GET", "POST", "PUT", "DELETE", "runs",
        "try", "on", "error", "finally", "log", "len", "get", "type",
        "and", "or", "true", "false", "nil", "nothing", "yes", "no",
        "import", "from", "export", "module",
        "match", "when", "at", "least", "most",
        "store", "gather", "into", "where",
        "emit", "notify", "wait", "seconds", "minutes",
        "every", "event",
        "memory_new", "memory_add", "memory_summary",
        "tensor_random", "tensor_matmul", "tensor_gelu",
        "tensor_softmax", "tensor_add", "tensor_layer_norm",
        "tensor_embedding", "tensor_create", "tensor_ones",
        "tensor_save", "tensor_load", "tensor_transpose",
        "tensor_scale", "tensor_cross_entropy",
        "read_file", "write_file",
        "env", "now", "string", "int", "float",
        "sqrt", "pow", "sin", "cos", "exp",
        "floor", "ceil", "round", "abs", "random",
        "json_encode", "json_decode",
        "split", "join", "trim", "upper", "lower", "contains",
        "replace", "starts_with", "ends_with", "substr"
    ]

    repeat for each kw in keywords:
        if get(vocab, kw) is equal nil:
            set vocab[kw] to vocab_size
            set vocab_size to vocab_size + 1

    // Add common NC patterns
    set patterns to [
        "to ", "set ", "if ", "repeat ", "respond with",
        "ask AI to", "save as", "is above", "is below",
        "is equal", "is not", "for each", "while ",
        "append ", "log ", "otherwise:",
        "api:", "runs ", "GET /", "POST /",
        "configure:", "purpose:", "version ",
        ": ", ", ", ". ",
        "{", "}", "[", "]", "(", ")",
        "+", "-", "*", "/", "=", ":",
        " "
    ]
    repeat for each pat in patterns:
        if get(vocab, pat) is equal nil:
            set vocab[pat] to vocab_size
            set vocab_size to vocab_size + 1

    // Add individual characters from corpus
    repeat for each text in corpus_texts:
        set i to 0
        repeat while i is below len(text):
            set ch to substr(text, i, 1)
            if get(vocab, ch) is equal nil:
                set vocab[ch] to vocab_size
                set vocab_size to vocab_size + 1
            set i to i + 1

    log "Vocabulary built: " + string(vocab_size) + " tokens"
    respond with vocab_size

// ── Learn BPE Merges ────────────────────────────────────────

to learn_merges with corpus_texts, num_merges:
    purpose: "Learn BPE merge rules from NC source corpus"
    log "Learning " + string(num_merges) + " BPE merges..."

    // Count adjacent token pair frequencies
    set pair_freq to {}
    repeat for each text in corpus_texts:
        set words to split(text, " ")
        repeat for each word in words:
            if len(word) is above 1:
                set ci to 0
                repeat while ci is below len(word) - 1:
                    set pair to substr(word, ci, 1) + "|" + substr(word, ci + 1, 1)
                    set current to get(pair_freq, pair)
                    if current is equal nil:
                        set pair_freq[pair] to 1
                    otherwise:
                        set pair_freq[pair] to current + 1
                    set ci to ci + 1

    // Find and add top merges
    set learned_merges to []
    set mi to 0
    repeat while mi is below num_merges:
        set best to ""
        set best_count to 0
        repeat for each entry in pair_freq:
            if entry.value is above best_count:
                set best_count to entry.value
                set best to entry.key
        if best is not equal "":
            append best to learned_merges
            set merged to replace(best, "|", "")
            if get(vocab, merged) is equal nil:
                set vocab[merged] to vocab_size
                set vocab_size to vocab_size + 1
            set pair_freq[best] to 0
        set mi to mi + 1

    log "Learned " + string(len(learned_merges)) + " merges, vocab now: " + string(vocab_size)
    respond with learned_merges

// ── Encode ──────────────────────────────────────────────────

to encode with text:
    purpose: "Encode NC source text to token IDs"
    set tokens to [1]
    set i to 0

    repeat while i is below len(text):
        set best_match to ""
        set best_len to 0

        // Try longest match from vocabulary
        set try_len to 20
        if try_len is above len(text) - i:
            set try_len to len(text) - i
        repeat while try_len is above 0:
            set window to substr(text, i, try_len)
            set tid to get(vocab, window)
            if tid is not equal nil:
                set best_match to window
                set best_len to try_len
                set try_len to 0
            otherwise:
                set try_len to try_len - 1

        if best_len is above 0:
            set tid to get(vocab, best_match)
            if tid is not equal nil:
                append tid to tokens
            otherwise:
                append 3 to tokens
            set i to i + best_len
        otherwise:
            // Single character fallback
            set ch to substr(text, i, 1)
            set tid to get(vocab, ch)
            if tid is not equal nil:
                append tid to tokens
            otherwise:
                append 3 to tokens
            set i to i + 1

    append 2 to tokens
    respond with tokens

// ── Decode ──────────────────────────────────────────────────

to decode with token_ids:
    purpose: "Decode token IDs back to NC source text"
    set parts to []

    // Build reverse map
    set rev to {}
    repeat for each entry in vocab:
        set rev[string(entry.value)] to entry.key

    repeat for each tid in token_ids:
        if tid is not equal 0:
            if tid is not equal 1:
                if tid is not equal 2:
                    set token_str to get(rev, string(tid))
                    if token_str is not equal nil:
                        append token_str to parts
                    otherwise:
                        append "?" to parts

    respond with join(parts, "")

// ── Save / Load ─────────────────────────────────────────────

to save_tokenizer with path:
    purpose: "Save tokenizer to disk"
    set data to {
        "vocab": vocab,
        "vocab_size": vocab_size
    }
    store data into path
    log "Tokenizer saved to " + path + " (" + string(vocab_size) + " tokens)"
    respond with true

to load_tokenizer with path:
    purpose: "Load tokenizer from disk"
    gather data from path
    if data is not equal nil:
        set vocab to data.vocab
        set vocab_size to data.vocab_size
        log "Tokenizer loaded: " + string(vocab_size) + " tokens"
        respond with true
    otherwise:
        log "Failed to load tokenizer from " + path
        respond with false

// ── Info ─────────────────────────────────────────────────────

to tokenizer_info:
    respond with {
        "name": "NC BPE Tokenizer",
        "version": "2.0.0",
        "vocab_size": vocab_size
    }

api:
    POST /encode        runs encode
    POST /decode        runs decode
    POST /vocab/build   runs build_vocab
    POST /vocab/save    runs save_tokenizer
    POST /vocab/load    runs load_tokenizer
    GET  /info          runs tokenizer_info

//  NC Transformer Model — Written 100% in NC Language
//
//  A real decoder-only transformer (GPT-style) using NC's
//  native tensor operations.
//
//  Architecture:
//    - Token embedding table (vocab_size × embed_dim)
//    - Positional encoding (max_seq × embed_dim)
//    - N transformer blocks:
//        - Multi-head self-attention (Q, K, V projections)
//        - Feed-forward network (up_proj → GELU → down_proj)
//        - Layer normalization (pre-norm)
//        - Residual connections
//    - Language model head (embed_dim → vocab_size)
//
//  Usage:
//    nc nc-ai/nc/models/transformer.nc
// ═══════════════════════════════════════════════════════════

service "nc-transformer"
version "2.0.0"

// ── Initialize Model ────────────────────────────────────────

to init_model with config:
    purpose: "Initialize transformer with random weights"
    log "Initializing NC Transformer..."
    log "  vocab_size: " + string(config.vocab_size)
    log "  embed_dim:  " + string(config.embed_dim)
    log "  num_heads:  " + string(config.num_heads)
    log "  num_layers: " + string(config.num_layers)
    log "  ff_dim:     " + string(config.ff_dim)

    // Token embedding table: vocab_size × embed_dim
    set model_config to config
    set token_embedding to tensor_random(config.vocab_size, config.embed_dim)
    set pos_embedding to tensor_random(config.max_seq_len, config.embed_dim)

    // Initialize transformer layers
    set layers to []
    set layer_idx to 0
    repeat while layer_idx is below config.num_layers:
        set layer to {
            "attn_q_weight": tensor_random(config.embed_dim, config.embed_dim),
            "attn_k_weight": tensor_random(config.embed_dim, config.embed_dim),
            "attn_v_weight": tensor_random(config.embed_dim, config.embed_dim),
            "attn_o_weight": tensor_random(config.embed_dim, config.embed_dim),
            "attn_q_bias": tensor_create(1, config.embed_dim),
            "attn_k_bias": tensor_create(1, config.embed_dim),
            "attn_v_bias": tensor_create(1, config.embed_dim),
            "attn_o_bias": tensor_create(1, config.embed_dim),
            "attn_norm_gamma": tensor_ones(1, config.embed_dim),
            "attn_norm_beta": tensor_create(1, config.embed_dim),
            "ff_up_weight": tensor_random(config.embed_dim, config.ff_dim),
            "ff_up_bias": tensor_create(1, config.ff_dim),
            "ff_down_weight": tensor_random(config.ff_dim, config.embed_dim),
            "ff_down_bias": tensor_create(1, config.embed_dim),
            "ff_norm_gamma": tensor_ones(1, config.embed_dim),
            "ff_norm_beta": tensor_create(1, config.embed_dim)
        }
        append layer to layers
        set layer_idx to layer_idx + 1

    // Final layer norm
    set final_norm_gamma to tensor_ones(1, config.embed_dim)
    set final_norm_beta to tensor_create(1, config.embed_dim)

    // LM head: embed_dim → vocab_size
    set lm_head to tensor_random(config.embed_dim, config.vocab_size)

    set model_initialized to true
    log "Model initialized: " + string(config.num_layers) + " layers"
    respond with true

// ── Self-Attention ──────────────────────────────────────────

to self_attention with x, layer:
    purpose: "Multi-head self-attention with causal masking"

    // Pre-norm
    set normed to tensor_layer_norm(x, layer.attn_norm_gamma, layer.attn_norm_beta)

    // Q, K, V projections
    set Q to tensor_matmul(normed, layer.attn_q_weight)
    set Q to tensor_add_bias(Q, layer.attn_q_bias)
    set K to tensor_matmul(normed, layer.attn_k_weight)
    set K to tensor_add_bias(K, layer.attn_k_bias)
    set V to tensor_matmul(normed, layer.attn_v_weight)
    set V to tensor_add_bias(V, layer.attn_v_bias)

    // Attention scores: Q × K^T / sqrt(d_k)
    set K_T to tensor_transpose(K)
    set scores to tensor_matmul(Q, K_T)
    set d_k to model_config.embed_dim / model_config.num_heads
    set scale to sqrt(d_k)
    set scores to tensor_scale(scores, 1.0 / scale)

    // Causal mask
    set mask to tensor_causal_mask(len(x))
    set scores to tensor_add(scores, mask)

    // Softmax → weighted sum
    set weights to tensor_softmax(scores)
    set context to tensor_matmul(weights, V)

    // Output projection + residual
    set out to tensor_matmul(context, layer.attn_o_weight)
    set out to tensor_add_bias(output, layer.attn_o_bias)
    set out to tensor_add(x, output)

    respond with out

// ── Feed-Forward Network ────────────────────────────────────

to feed_forward with x, layer:
    purpose: "Two-layer FFN with GELU activation"

    // Pre-norm
    set normed to tensor_layer_norm(x, layer.ff_norm_gamma, layer.ff_norm_beta)

    // Up projection + GELU
    set hidden to tensor_matmul(normed, layer.ff_up_weight)
    set hidden to tensor_add_bias(hidden, layer.ff_up_bias)
    set hidden to tensor_gelu(hidden)

    // Down projection + residual
    set out to tensor_matmul(hidden, layer.ff_down_weight)
    set out to tensor_add_bias(output, layer.ff_down_bias)
    set out to tensor_add(x, output)

    respond with out

// ── Full Forward Pass ───────────────────────────────────────

to forward with token_ids:
    purpose: "Full transformer forward pass: token IDs → logits"
    set seq_len to len(token_ids)

    // Token embedding lookup
    set x to tensor_embedding(token_embedding, token_ids)

    // Add positional embedding
    set pos_ids to []
    set i to 0
    repeat while i is below seq_len:
        append i to pos_ids
        set i to i + 1
    set pos_emb to tensor_embedding(pos_embedding, pos_ids)
    set x to tensor_add(x, pos_emb)

    // Pass through transformer layers
    set layer_idx to 0
    repeat while layer_idx is below model_config.num_layers:
        set layer to get(layers, layer_idx)
        set x to self_attention(x, layer)
        set x to feed_forward(x, layer)
        set layer_idx to layer_idx + 1

    // Final layer norm
    set x to tensor_layer_norm(x, final_norm_gamma, final_norm_beta)

    // LM head: project to vocabulary logits
    set logits to tensor_matmul(x, lm_head)

    respond with logits

// ── Generation (Autoregressive) ──────────────────────────────

to generate with token_ids, max_new_tokens, temperature:
    purpose: "Generate tokens autoregressively"
    set generated to token_ids

    set step to 0
    repeat while step is below max_new_tokens:
        // Truncate to max_seq_len
        if len(generated) is above model_config.max_seq_len:
            set start to len(generated) - model_config.max_seq_len
            set generated to slice(generated, start, len(generated))

        // Forward pass
        set logits to forward(generated)

        // Get last token's logits
        set last_row_idx to len(logits) - 1
        set last_logits to get(logits, last_row_idx)

        // Apply temperature
        if temperature is above 0:
            set scaled to []
            repeat for each logit in last_logits:
                append logit / temperature to scaled
            set last_logits to scaled

        // Softmax to get probabilities
        set probs to tensor_softmax([last_logits])
        set probs to get(probs, 0)

        // Greedy sampling (argmax)
        set next_token to 0
        set best_prob to 0
        set i to 0
        repeat for each p in probs:
            if p is above best_prob:
                set best_prob to p
                set next_token to i
            set i to i + 1

        append next_token to generated

        // Stop on EOS token (id=2)
        if next_token is equal 2:
            set step to max_new_tokens

        set step to step + 1

    respond with generated

// ── Save / Load Model ────────────────────────────────────────

to save_model with path:

to build_vocab with corpus_texts:
    purpose: "Build vocabulary from NC source texts"
    set vocab to {"<PAD>": 0, "<BOS>": 1, "<EOS>": 2, "<UNK>": 3}
    set vocab_size to 4

    repeat for each kw in keywords:
        if get(vocab, kw) is equal nil:
            set vocab[kw] to vocab_size
            set vocab_size to vocab_size + 1
    set patterns to [": ", ", ", ". ", "  ", "    ", "{", "}", "[", "]", "(", ")", "+", "-", "*", "/", "=", ":", " ", "\n", "\"", "'", "_", "//"]
    repeat for each pat in patterns:
        if get(vocab, pat) is equal nil:
            set vocab[pat] to vocab_size
            set vocab_size to vocab_size + 1
    set chars to "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    set ci to 0
    repeat while ci is below len(chars):
        set ch to substr(chars, ci, 1)
        if get(vocab, ch) is equal nil:

// ── Encode ──────────────────────────────────────────────────

to encode with text:
    purpose: "Encode text to token IDs"
    set tokens to [1]
    set i to 0
    repeat while i is below len(text):
        set best_len to 0
        set try_len to 15
        if try_len is above len(text) - i:
            set try_len to len(text) - i
        repeat while try_len is above 0:
            set window to substr(text, i, try_len)
            set tid to get(vocab, window)
            if tid is not equal nil:
                set best_len to try_len
                set try_len to 0
            otherwise:
                set try_len to try_len - 1
        if best_len is above 0:
            set window to substr(text, i, best_len)
            append get(vocab, window) to tokens
            set i to i + best_len
        otherwise:
            set ch to substr(text, i, 1)
            set tid to get(vocab, ch)
            if tid is not equal nil:
                append tid to tokens
            otherwise:
                append 3 to tokens
            set i to i + 1
    append 2 to tokens
    respond with tokens

// ── Decode ──────────────────────────────────────────────────

to decode with token_ids:

    set parts to []
    set rev to {}
    repeat for each entry in vocab:
        set rev[string(entry.value)] to entry.key
    repeat for each tid in token_ids:
        if tid is not equal 0:
            if tid is not equal 1:
                if tid is not equal 2:

                    if s is not equal nil:
                        append s to parts

to init_model with config:
    purpose: "Initialize transformer"
    log "Init model: " + string(config.embed_dim) + "d, " + string(config.num_layers) + " layers"
    set model_config to config
    set token_embedding to tensor_random(config.vocab_size, config.embed_dim)
    set pos_embedding to tensor_random(config.max_seq_len, config.embed_dim)

    repeat while li is below config.num_layers:
        set layer to {"qw": tensor_random(config.embed_dim, config.embed_dim), "kw": tensor_random(config.embed_dim, config.embed_dim), "vw": tensor_random(config.embed_dim, config.embed_dim), "ow": tensor_random(config.embed_dim, config.embed_dim), "ng": tensor_ones(1, config.embed_dim), "nb": tensor_create(1, config.embed_dim), "uw": tensor_random(config.embed_dim, config.ff_dim), "dw": tensor_random(config.ff_dim, config.embed_dim), "fg": tensor_ones(1, config.embed_dim), "fb": tensor_create(1, config.embed_dim)}
        append layer to layers

    set final_ng to tensor_ones(1, config.embed_dim)

    set lm_head to tensor_random(config.embed_dim, config.vocab_size)
    log "Model ready"
    respond with true

// ── Forward Pass ────────────────────────────────────────────

to forward with token_ids:
    purpose: "Transformer forward pass"
    set seq_len to len(token_ids)
    set x to tensor_embedding(token_embedding, token_ids)
    set pos_ids to []
    set i to 0
    repeat while i is below seq_len:
        append i to pos_ids
        set i to i + 1
    set pe to tensor_embedding(pos_embedding, pos_ids)
    set x to tensor_add(x, pe)
    set li to 0
    repeat while li is below model_config.num_layers:
        set L to layers[li]
        set n to tensor_layer_norm(x, L.ng, L.nb)
        set Q to tensor_matmul(n, L.qw)
        set K to tensor_matmul(n, L.kw)
        set V to tensor_matmul(n, L.vw)
        set KT to tensor_transpose(K)
        set sc to tensor_matmul(Q, KT)
        set dk to model_config.embed_dim / model_config.num_heads
        set sc to tensor_scale(sc, 1.0 / sqrt(dk))
        set mask to tensor_causal_mask(seq_len)
        set sc to tensor_add(sc, mask)
        set w to tensor_softmax(sc)
        set ctx to tensor_matmul(w, V)
        set o to tensor_matmul(ctx, L.ow)
        set x to tensor_add(x, o)
        set n2 to tensor_layer_norm(x, L.fg, L.fb)
        set h to tensor_matmul(n2, L.uw)
        set h to tensor_gelu(h)
        set h to tensor_matmul(h, L.dw)
        set x to tensor_add(x, h)
        set li to li + 1
    set x to tensor_layer_norm(x, final_ng, final_nb)
    set logits to tensor_matmul(x, lm_head)
    respond with logits

// ── Compute Loss ────────────────────────────────────────────

to compute_loss with token_batch, target_batch:
    purpose: "Cross-entropy loss on batch"
    set total to 0
    set bs to len(token_batch)
    set i to 0
    repeat while i is below bs:
        set logits to forward(token_batch[i])
        set loss to tensor_cross_entropy(logits, target_batch[i])
        set total to total + loss
        set i to i + 1
    respond with total / bs

    purpose: "Tokenize and prepare batch"

            append slice(tokens, 0, len(tokens) - 1) to inp
            append slice(tokens, 1, len(tokens)) to tgt
        set i to i + 1
    respond with {"inputs": inp, "targets": tgt}

to train_epoch with train_pairs, bs, max_len, step:
    purpose: "Train one epoch"
    set epoch_loss to 0
    set batches to 0
    set pos to 0
    repeat while pos is below len(train_pairs):

        set batch to prepare_batch(slice(train_pairs, pos, end), end - pos, max_len)
        set loss to compute_loss(batch.inputs, batch.targets)
        set epoch_loss to epoch_loss + loss
        set batches to batches + 1
        set step to step + 1
        set pos to end
    respond with {"loss": epoch_loss / batches, "step": step}
