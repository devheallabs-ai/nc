#!/usr/bin/env bash
# start.sh — DevHeal Labs website local dev helper
set -euo pipefail

# Colors
CYAN='\033[0;36m'
PURPLE='\033[0;35m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BOLD='\033[1m'
RESET='\033[0m'

PORT=9000
DIR="$(cd "$(dirname "$0")" && pwd)"

banner() {
  echo ""
  echo -e "${CYAN}${BOLD}  DevHeal Labs AI${RESET}"
  echo -e "${PURPLE}  devheallabs.in${RESET}"
  echo ""
}

serve() {
  banner
  echo -e "${GREEN}Starting local preview server...${RESET}"
  echo ""
  echo -e "  ${BOLD}URL:${RESET}  http://localhost:${PORT}"
  echo -e "  ${BOLD}Dir:${RESET}  ${DIR}"
  echo ""
  echo -e "${YELLOW}Press Ctrl+C to stop.${RESET}"
  echo ""
  cd "$DIR"
  python3 -m http.server "$PORT"
}

deploy_instructions() {
  banner
  echo -e "${BOLD}Deploy to GitHub Pages${RESET}"
  echo ""
  echo -e "  ${CYAN}1.${RESET} Clone the deployment repo (if not already):"
  echo -e "     ${PURPLE}git clone git@github.com:DevHealLabs/devheallabs.github.io.git${RESET}"
  echo ""
  echo -e "  ${CYAN}2.${RESET} Copy files to the deployment repo:"
  echo -e "     ${PURPLE}cp ${DIR}/index.html ${DIR}/404.html ${DIR}/CNAME <deploy-repo>/${RESET}"
  echo ""
  echo -e "  ${CYAN}3.${RESET} Commit and push:"
  echo -e "     ${PURPLE}cd <deploy-repo>${RESET}"
  echo -e "     ${PURPLE}git add -A${RESET}"
  echo -e "     ${PURPLE}git commit -m \"Update website\"${RESET}"
  echo -e "     ${PURPLE}git push origin main${RESET}"
  echo ""
  echo -e "${GREEN}GitHub Pages will deploy automatically within a few minutes.${RESET}"
  echo ""
}

case "${1:-}" in
  deploy)
    deploy_instructions
    ;;
  help|--help|-h)
    banner
    echo -e "${BOLD}Usage:${RESET}"
    echo -e "  ${CYAN}./start.sh${RESET}          Preview website locally on port ${PORT}"
    echo -e "  ${CYAN}./start.sh deploy${RESET}   Show deployment instructions"
    echo -e "  ${CYAN}./start.sh help${RESET}     Show this help message"
    echo ""
    ;;
  *)
    serve
    ;;
esac

#!/bin/bash
# ══════════════════════════════════════════════════════════════════
#  HiveANT — Docker Deploy
#  Usage: ./scripts/deploy-docker.sh [build|up|down|logs|pull-model]
# ══════════════════════════════════════════════════════════════════

set -e

ACTION="${1:-up}"
COMPOSE="docker compose"
command -v docker-compose &>/dev/null && COMPOSE="docker-compose"

case "$ACTION" in
    build)
        echo "Building HiveANT Docker image..."
        $COMPOSE build --no-cache
        echo "[ok] Build complete"
        ;;
    up)
        echo "Starting HiveANT stack..."
        $COMPOSE up -d --build
        echo ""
        echo "Waiting for services..."
        sleep 5

        echo "Pulling AI model (this may take a few minutes)..."
        $COMPOSE exec ollama ollama pull llama3 2>/dev/null || echo "[!!] Model pull failed — pull manually: docker compose exec ollama ollama pull llama3"

        echo ""
        echo "Initializing kernel..."
        curl -s -X POST http://localhost:7700/kernel/init 2>/dev/null | python3 -m json.tool 2>/dev/null || echo "[!!] Kernel init failed — server may still be starting"

        echo ""
        echo "Health check..."
        curl -s http://localhost:7700/health/deep 2>/dev/null | python3 -m json.tool 2>/dev/null || echo "[!!] Health check pending"

        echo ""
        echo "═══════════════════════════════════════════"
        echo "  HiveANT is running!"
        echo "  Dashboard:  http://localhost:7700"
        echo "  Ollama:     http://localhost:11434"
        echo "  Stop:       ./scripts/deploy-docker.sh down"
        echo "  Logs:       ./scripts/deploy-docker.sh logs"
        echo "═══════════════════════════════════════════"
        ;;
    down)
        echo "Stopping HiveANT stack..."
        $COMPOSE down
        echo "[ok] Stopped"
        ;;
    down-clean)
        echo "Stopping HiveANT and removing volumes..."
        $COMPOSE down -v
        echo "[ok] Stopped and cleaned"
        ;;
    logs)
        $COMPOSE logs -f hiveant
        ;;
    pull-model)
        MODEL="${2:-llama3}"
        echo "Pulling model: $MODEL"
        $COMPOSE exec ollama ollama pull "$MODEL"
        echo "[ok] Model $MODEL ready"
        ;;
    status)
        $COMPOSE ps
        echo ""
        curl -s http://localhost:7700/health/deep 2>/dev/null | python3 -m json.tool 2>/dev/null || echo "HiveANT not responding"
        ;;
    *)
        echo "Usage: $0 {build|up|down|down-clean|logs|pull-model|status}"
        exit 1
        ;;
esac


// FILE: ./hiveant/scripts/deploy-k8s.sh

#!/bin/bash
# ══════════════════════════════════════════════════════════════════
#  HiveANT — Kubernetes Deploy
#  Usage: ./scripts/deploy-k8s.sh [apply|delete|status|build|port-forward]
# ══════════════════════════════════════════════════════════════════

set -e

ACTION="${1:-apply}"
NAMESPACE="hiveant"
IMAGE_TAG="${IMAGE_TAG:-latest}"
REGISTRY="${REGISTRY:-}"

case "$ACTION" in
    build)
        echo "Building HiveANT Docker image..."
        if [ -n "$REGISTRY" ]; then
            docker build -t "$REGISTRY/hiveant:$IMAGE_TAG" .
            echo "Pushing to $REGISTRY..."
            docker push "$REGISTRY/hiveant:$IMAGE_TAG"
            echo "[ok] Image pushed: $REGISTRY/hiveant:$IMAGE_TAG"
        else
            docker build -t "hiveant:$IMAGE_TAG" .
            echo "[ok] Image built: hiveant:$IMAGE_TAG"
            echo "     Set REGISTRY=your-registry.io to push"
        fi
        ;;
    apply)
        echo "Deploying HiveANT to Kubernetes..."
        echo ""

        if ! kubectl get namespace $NAMESPACE &>/dev/null; then
            echo "Creating namespace..."
            kubectl apply -f k8s/namespace.yaml
        fi

        echo "Applying manifests..."
        kubectl apply -k k8s/

        echo ""
        echo "Waiting for Ollama..."
        kubectl -n $NAMESPACE rollout status deployment/ollama --timeout=120s 2>/dev/null || echo "[!!] Ollama still starting"

        echo ""
        echo "Waiting for HiveANT..."
        kubectl -n $NAMESPACE rollout status deployment/hiveant --timeout=120s 2>/dev/null || echo "[!!] HiveANT still starting"

        echo ""
        echo "Pulling AI model..."
        sleep 5
        kubectl -n $NAMESPACE exec deploy/ollama -- ollama pull llama3 2>/dev/null || echo "[!!] Pull model manually: kubectl -n $NAMESPACE exec deploy/ollama -- ollama pull llama3"

        echo ""
        echo "═══════════════════════════════════════════"
        echo "  HiveANT deployed to Kubernetes!"
        echo ""
        echo "  Port forward:  ./scripts/deploy-k8s.sh port-forward"
        echo "  Status:        ./scripts/deploy-k8s.sh status"
        echo "  Dashboard:     http://localhost:7700 (after port-forward)"
        echo "  Delete:        ./scripts/deploy-k8s.sh delete"
        echo "═══════════════════════════════════════════"
        ;;
    delete)
        echo "Removing HiveANT from Kubernetes..."
        kubectl delete -k k8s/ --ignore-not-found
        echo "[ok] Removed"
        ;;
    delete-all)
        echo "Removing HiveANT and all data..."
        kubectl delete -k k8s/ --ignore-not-found
        kubectl delete namespace $NAMESPACE --ignore-not-found
        echo "[ok] Removed everything"
        ;;
    status)
        echo "═══ Namespace ═══"
        kubectl -n $NAMESPACE get all 2>/dev/null || echo "Namespace $NAMESPACE not found"
        echo ""
        echo "═══ PVCs ═══"
        kubectl -n $NAMESPACE get pvc 2>/dev/null || true
        echo ""
        echo "═══ HPA ═══"
        kubectl -n $NAMESPACE get hpa 2>/dev/null || true
        echo ""
        echo "═══ Pods ═══"
        kubectl -n $NAMESPACE get pods -o wide 2>/dev/null || true
        ;;
    port-forward)
        echo "Port forwarding HiveANT to localhost:7700..."
        echo "Dashboard: http://localhost:7700"
        echo "Press Ctrl+C to stop"
        kubectl -n $NAMESPACE port-forward svc/hiveant 7700:7700
        ;;
    logs)
        kubectl -n $NAMESPACE logs -f deploy/hiveant
        ;;
    init)
        echo "Initializing HiveANT kernel..."
        kubectl -n $NAMESPACE exec deploy/hiveant -- curl -s -X POST http://localhost:7700/kernel/init | python3 -m json.tool 2>/dev/null || echo "Use port-forward first"
        ;;
    *)
        echo "Usage: $0 {build|apply|delete|delete-all|status|port-forward|logs|init}"
        exit 1
        ;;
esac

// FILE: ./hiveant/start.sh

#!/bin/bash
# ══════════════════════════════════════════════════════════════════
#  HiveANT — Main Entry Point
#  Swarm Autonomous AI Operating System
#
#  Usage:
#    ./start.sh                 Start HiveANT server (default)
#    ./start.sh setup           First-time setup
#    ./start.sh test            Run all tests
#    ./start.sh docker          Build and run with Docker
#    ./start.sh docker-compose  Start with docker-compose (includes Ollama)
#    ./start.sh k8s             Deploy to Kubernetes
#    ./start.sh status          Health check
#    ./start.sh stop            Stop running services
# ══════════════════════════════════════════════════════════════════

set -e

# ── Colors ────────────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC_COLOR='\033[0m' # No Color

# ── Helpers ───────────────────────────────────────────────────────
info()    { echo -e "${BLUE}[info]${NC_COLOR}  $1"; }
success() { echo -e "${GREEN}[ok]${NC_COLOR}    $1"; }
warn()    { echo -e "${YELLOW}[warn]${NC_COLOR}  $1"; }
error()   { echo -e "${RED}[error]${NC_COLOR} $1"; }

banner() {
    echo ""
    echo -e "${CYAN}${BOLD}"
    echo "  ╔═══════════════════════════════════════════════════╗"
    echo "  ║         HiveANT — Swarm Autonomous AI OS         ║"
    echo "  ║     Ant Colony + Bee Colony Intelligence          ║"
    echo "  ╚═══════════════════════════════════════════════════╝"
    echo -e "${NC_COLOR}"
}

# ── Change to script directory (project root) ────────────────────
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$SCRIPT_DIR"

# ── Detect NC binary ─────────────────────────────────────────────
find_nc() {
    # 1. Check PATH first
    if command -v nc &> /dev/null; then
        # Make sure it is the NC engine, not netcat
        if nc --version 2>&1 | grep -qi "notation\|nc-lang\|NC Engine"; then
            NC_BIN="$(command -v nc)"
            return 0
        fi
    fi

    # 2. Check sibling build directory (nc-main layout)
    local PARENT_DIR="$(dirname "$SCRIPT_DIR")"
    if [ -f "$PARENT_DIR/nc/build/nc" ]; then
        NC_BIN="$PARENT_DIR/nc/build/nc"
        return 0
    fi

    # 3. Check common local build paths
    local SEARCH_PATHS=(
        "$HOME/Documents/nc-main/nc/build/nc"
        "$HOME/Documents/notation-as-code/nc/build/nc"
        "$HOME/nc-main/nc/build/nc"
        "/usr/local/bin/nc"
    )
    for p in "${SEARCH_PATHS[@]}"; do
        if [ -f "$p" ]; then
            NC_BIN="$p"
            return 0
        fi
    done

    return 1
}

# ── Required environment variables ───────────────────────────────
setup_env() {
    export NC_ALLOW_EXEC=1
    export NC_ALLOW_FILE_WRITE=1

    # Load config/.env if it exists
    if [ -f config/.env ]; then
        set -a
        source config/.env
        set +a
        info "Loaded config/.env"
    fi
}

# ── Create required directories ──────────────────────────────────
ensure_dirs() {
    mkdir -p \
        incidents knowledge docs \
        memory/pheromone/edges memory/twins/services memory/tasks memory/causal \
        memory/knowledge_graph/entities memory/knowledge_graph/relations \
        agents_state/queue agents_state/clusters agents_state/audit \
        agents_state/bus agents_state/plugins agents_state/users agents_state/orgs \
        sandbox_envs simulation/results
}

# ══════════════════════════════════════════════════════════════════
#  Commands
# ══════════════════════════════════════════════════════════════════

cmd_start() {
    banner
    setup_env
    ensure_dirs

    if ! find_nc; then
        error "NC Engine not found."
        echo ""
        echo "  Install options:"
        echo "    1. Build from source:  cd ../nc && make"
        echo "    2. Add to PATH:        export PATH=\$PATH:/path/to/nc/build"
        echo ""
        exit 1
    fi

    success "NC Engine: $NC_BIN"
    info "Port: ${HIVEANT_PORT:-7700}"
    info "Dashboard: http://localhost:${HIVEANT_PORT:-7700}"
    echo ""

    "$NC_BIN" serve server.nc
}

cmd_setup() {
    banner
    info "Running first-time setup..."
    echo ""

    # 1. Check NC Engine
    if find_nc; then
        success "NC Engine found: $NC_BIN"
    else
        warn "NC Engine not found. Build it first:"
        echo "       cd ../nc && make"
    fi

    # 2. Check Ollama
    if command -v ollama &> /dev/null; then
        success "Ollama installed"
        if curl -s http://localhost:11434/api/tags > /dev/null 2>&1; then
            success "Ollama running"
        else
            warn "Ollama not running. Start with: ollama serve"
        fi
    else
        warn "Ollama not installed (needed for local AI)"
        echo "       Install: brew install ollama (macOS) or see https://ollama.ai"
    fi

    # 3. Check Docker
    if command -v docker &> /dev/null; then
        success "Docker installed"
    else
        info "Docker not installed (optional, for containerized deployment)"
    fi

    # 4. Check kubectl
    if command -v kubectl &> /dev/null; then
        success "kubectl installed"
    else
        info "kubectl not installed (optional, for Kubernetes deployment)"
    fi

    # 5. Create config
    if [ ! -f config/.env ]; then
        mkdir -p config
        if [ -f config/.env.example ]; then
            cp config/.env.example config/.env
            success "Created config/.env from template"
            warn "Edit config/.env to set your AI model URL and key"
        else
            warn "config/.env.example not found, skipping config copy"
        fi
    else
        success "config/.env already exists"
    fi

    # 6. Create directories
    ensure_dirs
    success "Data directories created"

    echo ""
    echo -e "${GREEN}${BOLD}  Setup complete!${NC_COLOR}"
    echo ""
    echo "  Next steps:"
    echo "    1. Edit config/.env with your AI model settings"
    echo "    2. Run: ./start.sh"
    echo "    3. Open: http://localhost:${HIVEANT_PORT:-7700}"
    echo ""
}

cmd_test() {
    banner
    setup_env
    ensure_dirs

    if ! find_nc; then
        error "NC Engine not found. Cannot run tests."
        exit 1
    fi

    info "Running HiveANT test suite..."
    echo ""

    local FAIL=0

    # NC unit tests
    for test_file in tests/test_*.nc; do
        if [ -f "$test_file" ]; then
            info "Running $test_file..."
            if "$NC_BIN" test "$test_file"; then
                success "$test_file passed"
            else
                error "$test_file FAILED"
                FAIL=1
            fi
            echo ""
        fi
    done

    # Shell-based tests
    for test_file in tests/test_*.sh; do
        if [ -f "$test_file" ]; then
            info "Running $test_file..."
            if bash "$test_file"; then
                success "$test_file passed"
            else
                error "$test_file FAILED"
                FAIL=1
            fi
            echo ""
        fi
    done

    if [ $FAIL -eq 0 ]; then
        echo ""
        success "All tests passed."
    else
        echo ""
        error "Some tests failed."
        exit 1
    fi
}

    echo ""

        success "HiveANT server is running"
        echo ""
        info "Health details:"
        curl -s http://localhost:${HIVEANT_PORT:-7700}/health/deep 2>/dev/null | python3 -m json.tool 2>/dev/null || echo "  (could not parse response)"
    else
        warn "HiveANT server is not responding on port ${HIVEANT_PORT:-7700}"
    fi

    echo ""

    # Check Docker containers
    if command -v docker &> /dev/null; then
        if docker ps --filter "name=hiveant" --format "{{.Names}}: {{.Status}}" 2>/dev/null | grep -q hiveant; then
            info "Docker containers:"
            docker ps --filter "name=hiveant" --format "  {{.Names}}: {{.Status}}"
        fi
    fi

    # Check Kubernetes
    if command -v kubectl &> /dev/null; then
        if kubectl -n hiveant get pods 2>/dev/null | grep -q hiveant; then
            echo ""
            info "Kubernetes pods:"
            kubectl -n hiveant get pods 2>/dev/null | sed 's/^/  /'
        fi
    fi
}

cmd_stop() {
    banner
    info "Stopping HiveANT services..."
    echo ""

    # Stop Docker container
    if command -v docker &> /dev/null; then
        if docker ps -q --filter "name=hiveant" 2>/dev/null | grep -q .; then
            info "Stopping Docker containers..."
            COMPOSE="docker compose"
            command -v docker-compose &>/dev/null && COMPOSE="docker-compose"

            if $COMPOSE ps 2>/dev/null | grep -q hiveant; then
                $COMPOSE down
                success "Docker Compose stack stopped"
            else
                docker stop hiveant 2>/dev/null && docker rm hiveant 2>/dev/null
                success "Docker container stopped"
            fi
        else
            info "No Docker containers running"
        fi
    fi

    # Stop local process
    if pgrep -f "nc serve server.nc" > /dev/null 2>&1; then
        info "Stopping local NC server..."
        pkill -f "nc serve server.nc" 2>/dev/null || true
        success "Local server stopped"
    else
        info "No local server process found"
    fi

    echo ""
    success "All HiveANT services stopped"
}

#!/bin/bash
# ══════════════════════════════════════════════════════════════════
#  HiveANT — Performance & Load Tests
#  Usage: ./tests/test_performance.sh [http://localhost:7700]
#
#  Tests: Response time, throughput, concurrent load, memory
# ══════════════════════════════════════════════════════════════════

BASE="${1:-http://localhost:7700}"
green='\033[0;32m'; red='\033[0;31m'; yellow='\033[0;33m'; cyan='\033[0;36m'; nc='\033[0m'

echo ""
echo -e "  ${cyan}HiveANT Performance Tests${nc}"
echo "  Base: $BASE"
echo "  ════════════════════════════════"

# ── 1. Response Time Benchmarks ─────────────────────────────────
echo ""
echo -e "${yellow}1. Response Time (single request)${nc}"

bench() {
    local name="$1" method="$2" path="$3" body="$4"
    local start=$(date +%s%N 2>/dev/null || python3 -c "import time;print(int(time.time()*1000000000))")
    if [ "$method" = "GET" ]; then
        curl -sf "$BASE$path" -o /dev/null 2>/dev/null
    else
        curl -sf -X POST "$BASE$path" -H "Content-Type: application/json" -d "$body" -o /dev/null 2>/dev/null
    fi
    local end=$(date +%s%N 2>/dev/null || python3 -c "import time;print(int(time.time()*1000000000))")
    local ms=$(( (end - start) / 1000000 ))
    if [ $ms -lt 100 ]; then
        echo -e "  ${green}${ms}ms${nc}  $name"
    elif [ $ms -lt 500 ]; then
        echo -e "  ${yellow}${ms}ms${nc}  $name"
    else
        echo -e "  ${red}${ms}ms${nc}  $name"
    fi
}

bench "GET /health" GET "/health"
bench "GET /health/deep" GET "/health/deep"
bench "GET /kernel/status" GET "/kernel/status"
bench "GET /agents" GET "/agents"
bench "GET /pheromone/status" GET "/pheromone/status"
bench "GET /knowledge" GET "/knowledge"
bench "GET /incidents" GET "/incidents"
bench "GET /auth/status" GET "/auth/status"
bench "POST /agents/spawn" POST "/agents/spawn" '{"agent_type":"detection","agent_id":"perf-test-001"}'
bench "POST /pheromone/add" POST "/pheromone/add" '{"from_node":"perf_a","to_node":"perf_b","score":1.0}'
bench "POST /auth/login" POST "/auth/login" '{"username":"admin","password":"admin123"}'

# ── 2. Throughput Test ──────────────────────────────────────────
echo ""
echo -e "${yellow}2. Throughput (100 sequential requests)${nc}"

throughput() {
    local name="$1" path="$2" count="${3:-100}"
    local start=$(date +%s)
    for i in $(seq 1 $count); do
        curl -sf "$BASE$path" -o /dev/null 2>/dev/null
    done
    local end=$(date +%s)
    local dur=$((end - start))
    if [ $dur -eq 0 ]; then dur=1; fi
    local rps=$((count / dur))
    echo -e "  ${green}${rps} req/s${nc}  $name ($count requests in ${dur}s)"
}

throughput "GET /health" "/health" 100
throughput "GET /kernel/status" "/kernel/status" 50

# ── 3. Concurrent Load Test ─────────────────────────────────────
echo ""
echo -e "${yellow}3. Concurrent Load (10 parallel requests)${nc}"

concurrent() {
    local name="$1" path="$2" concurrency="${3:-10}"
    local start=$(date +%s%N 2>/dev/null || python3 -c "import time;print(int(time.time()*1000000000))")
    for i in $(seq 1 $concurrency); do

    local ms=$(( (end - start) / 1000000 ))
    echo -e "  ${green}${ms}ms${nc}  $name ($concurrency concurrent)"
}

        result=$(curl -sf -X POST "$BASE$path" -H "Content-Type: application/json" -d "$body" 2>/dev/null || echo '{"error":"connection_failed"}')
    fi
    if echo "$result" | grep -q "$expect"; then
        echo -e "  ${green}PASS${nc} $desc"
        PASS=$((PASS + 1))
    else
        echo -e "  ${red}FAIL${nc} $desc (expected: $expect)"
        FAIL=$((FAIL + 1))
    fi
}

success() { echo -e "${GREEN}${BOLD}[OK]${RESET} $*"; }
warn()    { echo -e "${YELLOW}[WARN]${RESET} $*"; }

do_generate() {

do_status() {

    echo -e "  Providers:"

#!/usr/bin/env bash
# ═══════════════════════════════════════════════════════════
#  NC AI — Training Script
#  Builds the nc-ai engine (if needed) and trains on
#  the collected NC training data.
#
#  Usage:
#    ./train.sh                    # defaults: 10 epochs, dim 128
#    ./train.sh --epochs 20        # override epochs
#    ./train.sh --dim 256          # override model dimension
#    ./train.sh --lr 0.0005        # override learning rate
#    ./train.sh --layers 6         # override transformer layers
#
#  by DevHeal Labs AI
# ═══════════════════════════════════════════════════════════

set -euo pipefail

# ── Paths ───────────────────────────────────────────────────

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
ENGINE_DIR="$SCRIPT_DIR/engine"
BUILD_DIR="$ENGINE_DIR/build"
BINARY="$BUILD_DIR/nc-ai"
DATA_DIR="$SCRIPT_DIR/training_data"
MODEL_DIR="$SCRIPT_DIR/model"
MODEL_OUT="$MODEL_DIR/nc_model.bin"

# ── Defaults (overridable via flags) ───────────────────────

EPOCHS=10
DIM=128
LAYERS=4
LR=0.001

# ── Parse arguments ────────────────────────────────────────

while [[ $# -gt 0 ]]; do
    case "$1" in
        --epochs)  EPOCHS="$2";  shift 2 ;;
        --dim)     DIM="$2";     shift 2 ;;
        --layers)  LAYERS="$2";  shift 2 ;;
        --lr)      LR="$2";     shift 2 ;;
        --help|-h)
            echo "Usage: $0 [--epochs N] [--dim N] [--layers N] [--lr F]"
            exit 0
            ;;
        *) echo "Unknown flag: $1"; exit 1 ;;
    esac
done

# ── Colors ──────────────────────────────────────────────────

BOLD='\033[1m'
GREEN='\033[32m'
CYAN='\033[36m'
YELLOW='\033[33m'
RED='\033[31m'
RESET='\033[0m'

info()  { echo -e "${CYAN}[INFO]${RESET}  $*"; }
ok()    { echo -e "${GREEN}${BOLD}[OK]${RESET}    $*"; }
warn()  { echo -e "${YELLOW}[WARN]${RESET}  $*"; }
fail()  { echo -e "${RED}[FAIL]${RESET}  $*"; exit 1; }

# ═══════════════════════════════════════════════════════════
#  Step 1: Build nc-ai binary (if not present)
# ═══════════════════════════════════════════════════════════

echo ""
echo -e "${BOLD}  NC AI — Training Pipeline${RESET}"
echo "  ─────────────────────────────"
echo ""

if [[ -x "$BINARY" ]]; then
    ok "nc-ai binary found at $BINARY"
else
    info "nc-ai binary not found — building..."
    (cd "$ENGINE_DIR" && make all) || fail "Build failed. Check compiler output above."
    if [[ -x "$BINARY" ]]; then
        ok "Build successful: $BINARY"
    else
        fail "Build completed but binary not found at $BINARY"
    fi
fi

# ═══════════════════════════════════════════════════════════
#  Step 2: Verify training data
# ═══════════════════════════════════════════════════════════

info "Checking training data in $DATA_DIR ..."

NC_CODE="$DATA_DIR/nc_code.txt"
NCUI_CODE="$DATA_DIR/ncui_code.txt"
PROMPT_PAIRS="$DATA_DIR/prompt_pairs.txt"

missing=0
for f in "$NC_CODE" "$NCUI_CODE" "$PROMPT_PAIRS"; do
    if [[ ! -f "$f" ]]; then
        warn "Missing: $f"
        missing=1
    fi
done

if [[ $missing -eq 1 ]]; then
    fail "Training data is incomplete. Regenerate with the data collection script."
fi

# Count training samples
NC_SAMPLES=$(grep -c '^---$' "$NC_CODE" 2>/dev/null || echo 0)
NC_SAMPLES=$((NC_SAMPLES + 1))
NCUI_SAMPLES=$(grep -c '^---$' "$NCUI_CODE" 2>/dev/null || echo 0)
NCUI_SAMPLES=$((NCUI_SAMPLES + 1))
PROMPT_SAMPLES=$(grep -c '^PROMPT:' "$PROMPT_PAIRS" 2>/dev/null || echo 0)

NC_LINES=$(wc -l < "$NC_CODE" | tr -d ' ')
NCUI_LINES=$(wc -l < "$NCUI_CODE" | tr -d ' ')
TOTAL_SIZE=$(du -sh "$DATA_DIR" | cut -f1)

echo ""
echo "  Training Data Summary"
echo "  ─────────────────────────────"
echo "  NC code samples:     $NC_SAMPLES"
echo "  NC code lines:       $NC_LINES"
echo "  NCUI code samples:   $NCUI_SAMPLES"
echo "  NCUI code lines:     $NCUI_LINES"
echo "  Prompt-code pairs:   $PROMPT_SAMPLES"
echo "  Total data size:     $TOTAL_SIZE"
echo "  ─────────────────────────────"
echo ""

# ═══════════════════════════════════════════════════════════
#  Step 3: Run training
# ═══════════════════════════════════════════════════════════

mkdir -p "$MODEL_DIR"

info "Starting training..."
echo ""
echo "  Configuration"
echo "  ─────────────────────────────"
echo "  Epochs:      $EPOCHS"
echo "  Dimension:   $DIM"
echo "  Layers:      $LAYERS"
echo "  Learn rate:  $LR"
echo "  Data dir:    $DATA_DIR"
echo "  Output:      $MODEL_OUT"
echo "  ─────────────────────────────"
echo ""

START_TIME=$(date +%s)

"$BINARY" train \
    --data "$DATA_DIR" \
    --epochs "$EPOCHS" \
    --dim "$DIM" \
    --layers "$LAYERS" \
    --lr "$LR" \
    --output "$MODEL_OUT"

TRAIN_EXIT=$?
END_TIME=$(date +%s)
ELAPSED=$((END_TIME - START_TIME))

echo ""

# ═══════════════════════════════════════════════════════════
#  Step 4: Report results
# ═══════════════════════════════════════════════════════════

if [[ $TRAIN_EXIT -eq 0 ]]; then
    echo "  Training Results"
    echo "  ─────────────────────────────"
    echo "  Status:        SUCCESS"
    echo "  Duration:      ${ELAPSED}s"

    if [[ -f "$MODEL_OUT" ]]; then
        MODEL_SIZE=$(du -sh "$MODEL_OUT" | cut -f1)
        echo "  Model file:    $MODEL_OUT"
        echo "  Model size:    $MODEL_SIZE"
    fi

    echo "  Samples used:  $((NC_SAMPLES + NCUI_SAMPLES))"
    echo "  Prompt pairs:  $PROMPT_SAMPLES"
    echo "  ─────────────────────────────"
    echo ""
    ok "Training complete. Model saved to $MODEL_OUT"
else
    echo "  Training Results"
    echo "  ─────────────────────────────"
    echo "  Status:        FAILED (exit code $TRAIN_EXIT)"
    echo "  Duration:      ${ELAPSED}s"
    echo "  ─────────────────────────────"
    echo ""
    fail "Training failed. Check output above for errors."
fi

echo ""

    DIM=$(tput setaf 8 2>/dev/null || tput dim 2>/dev/null || echo "")
    RESET=$(tput sgr0)
else
    BOLD="" RED="" GREEN="" YELLOW="" BLUE="" CYAN="" DIM="" RESET=""
fi

info()    { echo "${CYAN}  [*]${RESET} $*"; }
success() { echo "${GREEN}  [+]${RESET} $*"; }
warn()    { echo "${YELLOW}  [!]${RESET} $*"; }
error()   { echo "${RED}  [x]${RESET} $*"; }
step()    { echo "${BOLD}${BLUE}  ==> ${RESET}${BOLD}$*${RESET}"; }

# ── Platform detection ────────────────────────────────────────────────
detect_platform() {
    local uname_s
    uname_s="$(uname -s 2>/dev/null || echo Unknown)"
    ARCH="$(uname -m 2>/dev/null || echo unknown)"

    case "$ARCH" in
        x86_64|amd64)   ARCH="x86_64" ;;
        aarch64|arm64)  ARCH="arm64" ;;
        armv7l)         ARCH="armv7" ;;
        i686|i386)      ARCH="x86" ;;
    esac

    case "$uname_s" in
        Darwin*)
            PLATFORM="macos"
            DISTRO="macos"
            BINARY_EXT=""
            ;;
        Linux*)
            PLATFORM="linux"
            BINARY_EXT=""
            # Detect distro
            if [ -f /etc/os-release ]; then
                . /etc/os-release
                case "$ID" in
                    ubuntu|debian|pop|linuxmint|elementary|zorin|kali)
                        DISTRO="debian" ;;
                    fedora|rhel|centos|rocky|alma|ol)
                        DISTRO="fedora" ;;
                    arch|manjaro|endeavouros|garuda)
                        DISTRO="arch" ;;
                    opensuse*|sles)
                        DISTRO="suse" ;;
                    alpine)
                        DISTRO="alpine" ;;
                    void)
                        DISTRO="void" ;;
                    *)
                        DISTRO="unknown" ;;
                esac
            elif [ -f /etc/debian_version ]; then
                DISTRO="debian"
            elif [ -f /etc/redhat-release ]; then
                DISTRO="fedora"
            else
                DISTRO="unknown"
            fi
            ;;
        MINGW*|MSYS*|CYGWIN*)
            PLATFORM="windows"
            DISTRO="msys2"
            BINARY_EXT=".exe"
            # Override install dir for Windows
            if [ -z "${NC_INSTALL_DIR:-}" ]; then
                INSTALL_DIR="${USERPROFILE:-$HOME}/bin"
            fi
            ;;
        *)
            PLATFORM="unknown"
            DISTRO="unknown"
            BINARY_EXT=""
            ;;
    esac
}

check_prerequisites() {
    step "Checking prerequisites"
    local missing=0

    if check_compiler; then
        success "C compiler: ${FOUND_CC} ($($FOUND_CC --version 2>&1 | head -1))"
    else
        error "C compiler: not found (need gcc, clang, or cc)"
        missing=1
    fi

    if check_make; then
        success "Build tool: ${FOUND_MAKE} ($($FOUND_MAKE --version 2>&1 | head -1))"
    else
        error "Build tool: make not found"
        missing=1
    fi

    if [ "$PLATFORM" != "windows" ]; then
        if pkg-config --exists libcurl 2>/dev/null || [ -f /usr/include/curl/curl.h ] || [ -f /usr/local/include/curl/curl.h ]; then
            success "libcurl: found"
        else
            warn "libcurl: not found (build may fail without it)"
        fi
    fi

    if [ "$missing" -eq 1 ]; then
        print_install_instructions
        exit 1
    fi
    echo ""
}

# ── Strategy 1: Download pre-built binary ────────────────────────────
try_prebuilt() {
    local asset_name="nc-${PLATFORM}-${ARCH}${BINARY_EXT}"
    local url="https://github.com/${REPO}/releases/download/v${NC_VERSION}/${asset_name}"

    step "Checking for pre-built binary"
    info "Looking for ${asset_name}..."

    local tmp_bin="/tmp/nc-installer-$$${BINARY_EXT}"
    local http_code=""

    if command -v curl >/dev/null 2>&1; then
        http_code=$(curl -fsSL -o "$tmp_bin" -w "%{http_code}" "$url" 2>/dev/null) || true
    elif command -v wget >/dev/null 2>&1; then
        if wget -q -O "$tmp_bin" "$url" 2>/dev/null; then
            http_code="200"
        fi
    fi

    if [ "${http_code:-}" = "200" ] && [ -s "$tmp_bin" ]; then
        chmod +x "$tmp_bin"
        if "$tmp_bin" version >/dev/null 2>&1; then
            success "Downloaded pre-built binary"
            INSTALL_BINARY="$tmp_bin"
            return 0
        fi
    fi

    rm -f "$tmp_bin"
    warn "No pre-built binary available, will build from source"
    echo ""
    return 1
}

# ── Strategy 2: Build from source ────────────────────────────────────
build_from_source() {
    step "Building NC from source"

    local script_dir
    script_dir="$(cd "$(dirname "${BASH_SOURCE[0]}")" 2>/dev/null && pwd)"
    local build_dir=""
    local src_dir=""
    local need_cleanup=false

    # Check if we're in a source checkout
    if [ -f "$script_dir/nc/Makefile" ]; then
        build_dir="$script_dir/nc"
        src_dir="$script_dir"
        info "Using local source at ${DIM}${build_dir}${RESET}"
    else
        # Download source
        info "Downloading NC source code..."
        local tmpdir
        tmpdir="$(mktemp -d)"
        need_cleanup=true

        if command -v git >/dev/null 2>&1; then
            git clone --depth 1 "https://github.com/${REPO}.git" "$tmpdir/nc" 2>/dev/null
        elif command -v curl >/dev/null 2>&1; then
            curl -sSL "https://github.com/${REPO}/archive/main.tar.gz" | tar xz -C "$tmpdir"
            mv "$tmpdir"/nc-*main* "$tmpdir/nc" 2>/dev/null || true
        elif command -v wget >/dev/null 2>&1; then
            wget -qO- "https://github.com/${REPO}/archive/main.tar.gz" | tar xz -C "$tmpdir"
            mv "$tmpdir"/nc-*main* "$tmpdir/nc" 2>/dev/null || true
        else
            error "Cannot download source: need git, curl, or wget"
            exit 1
        fi

        build_dir="$tmpdir/nc/nc"
        src_dir="$tmpdir/nc"

        if [ ! -f "$build_dir/Makefile" ]; then
            error "Makefile not found after download"
            rm -rf "$tmpdir"
            exit 1
        fi
    fi

    # Build
    info "Compiling (this may take a moment)..."
    cd "$build_dir"
    make clean >/dev/null 2>&1 || true

    if make 2>&1 | while IFS= read -r line; do
        # Show last few lines to indicate progress
        printf "  ${DIM}  %s${RESET}\n" "$line"
    done; then
        true
    fi

    local built_bin="build/nc${BINARY_EXT}"
    if [ ! -f "$built_bin" ]; then
        error "Build failed: ${built_bin} not produced"
        if $need_cleanup; then rm -rf "$tmpdir"; fi
        exit 1
    fi

    success "Build successful"

    local tmp_bin="/tmp/nc-installer-$$${BINARY_EXT}"
    cp "$built_bin" "$tmp_bin"
    chmod +x "$tmp_bin"
    INSTALL_BINARY="$tmp_bin"

    # Copy standard library
    if [ -d "$src_dir/Lib" ]; then
        INSTALL_LIB_SRC="$src_dir/Lib"
    fi

    if $need_cleanup; then
        # Keep the lib source around until install finishes
        CLEANUP_DIR="$tmpdir"
    fi
    echo ""
}

# ── Install binary ────────────────────────────────────────────────────
install_binary() {
    step "Installing NC"

    if [ -z "${INSTALL_BINARY:-}" ] || [ ! -f "${INSTALL_BINARY}" ]; then
        error "No binary to install"
        exit 1
    fi

    # Create install directory
    if [ ! -d "$INSTALL_DIR" ]; then
        if [ -w "$(dirname "$INSTALL_DIR")" ]; then
            mkdir -p "$INSTALL_DIR"
        else
            info "Creating ${INSTALL_DIR} (requires sudo)"
            sudo mkdir -p "$INSTALL_DIR"
        fi
    fi

    # Copy binary
    local dest="${INSTALL_DIR}/nc${BINARY_EXT}"
    if [ -w "$INSTALL_DIR" ]; then
        cp "$INSTALL_BINARY" "$dest"
    else
        info "Installing to ${INSTALL_DIR} (requires sudo)"
        sudo cp "$INSTALL_BINARY" "$dest"
        sudo chmod +x "$dest"
    fi
    success "Binary installed: ${dest}"

    # Copy standard library
    if [ -n "${INSTALL_LIB_SRC:-}" ] && [ -d "$INSTALL_LIB_SRC" ]; then
        if [ -w "$(dirname "$LIB_DIR")" ] || [ -w "$LIB_DIR" ] 2>/dev/null; then
            mkdir -p "$LIB_DIR"
            cp -r "$INSTALL_LIB_SRC/"* "$LIB_DIR/" 2>/dev/null || true
        else
            sudo mkdir -p "$LIB_DIR"
            sudo cp -r "$INSTALL_LIB_SRC/"* "$LIB_DIR/" 2>/dev/null || true
        fi
        success "Standard library installed: ${LIB_DIR}"
    fi

    # Windows/MSYS2: add to PATH via .bashrc or .bash_profile
    if [ "$PLATFORM" = "windows" ]; then
        local profile="${HOME}/.bashrc"
        if ! grep -q "nc.*bin" "$profile" 2>/dev/null; then
            echo "" >> "$profile"
            echo "# NC language" >> "$profile"
            echo "export PATH=\"${INSTALL_DIR}:\$PATH\"" >> "$profile"
            success "Added ${INSTALL_DIR} to PATH in ~/.bashrc"
        fi
    fi

    # Cleanup temp binary
    rm -f "$INSTALL_BINARY"
    if [ -n "${CLEANUP_DIR:-}" ]; then
        rm -rf "$CLEANUP_DIR"
    fi

    echo ""
}

# ── Verify installation ──────────────────────────────────────────────
verify_installation() {
    step "Verifying installation"

    export NC_ACCEPT_LICENSE=1
    local nc_bin="${INSTALL_DIR}/nc${BINARY_EXT}"

    if [ ! -f "$nc_bin" ]; then
        error "Binary not found at ${nc_bin}"
        exit 1
    fi

    if "$nc_bin" version >/dev/null 2>&1; then
        local ver_output
        ver_output="$("$nc_bin" version 2>&1 | head -1)"
        success "Verification passed: ${ver_output}"
    elif "$nc_bin" --version >/dev/null 2>&1; then
        local ver_output
        ver_output="$("$nc_bin" --version 2>&1 | head -1)"
        success "Verification passed: ${ver_output}"
    else
        warn "Binary exists but version check returned non-zero (may still work)"
    fi
    echo ""
}

# ── Print success ────────────────────────────────────────────────────
print_success() {
    echo "${GREEN}  ╔═══════════════════════════════════════╗${RESET}"
    echo "${GREEN}  ║     NC installed successfully!        ║${RESET}"
    echo "${GREEN}  ╚═══════════════════════════════════════╝${RESET}"
    echo ""
    echo "  ${BOLD}Location:${RESET}  ${INSTALL_DIR}/nc${BINARY_EXT}"
    echo "  ${BOLD}Platform:${RESET}  ${PLATFORM}/${ARCH}"
    echo ""
    echo "  ${BOLD}Get started:${RESET}"
    echo "    ${CYAN}nc version${RESET}                Show version info"
    echo "    ${CYAN}nc \"show 42 + 8\"${RESET}          Run inline code"
    echo "    ${CYAN}nc run hello.nc${RESET}            Run a program"
    echo "    ${CYAN}nc serve app.nc${RESET}            Start HTTP server"
    echo "    ${CYAN}nc repl${RESET}                    Interactive REPL"
    echo ""
    echo "  ${BOLD}Other install methods:${RESET}"
    echo "    ${DIM}pip install nc-lang${RESET}        Python package"
    echo "    ${DIM}brew install nc${RESET}            macOS Homebrew"
    echo "    ${DIM}docker pull nc-lang/nc${RESET}     Docker"
    echo ""

    if [ "$PLATFORM" = "windows" ]; then
        warn "Restart your terminal for PATH changes to take effect."
        echo ""
    fi
}

# ══════════════════════════════════════════════════════════════════════
#  Main
# ══════════════════════════════════════════════════════════════════════
main() {
    detect_platform
    print_banner
    require_license_acceptance
    write_consent_marker
    echo ""
    check_prerequisites

    # Try pre-built binary first, fall back to source build
    if ! try_prebuilt; then
        build_from_source
    fi

    install_binary
    verify_installation
    print_success
}

main "$@"


// FILE: ./nc-lang/scripts/deploy-websites.sh

#!/usr/bin/env bash
#
# deploy-websites.sh
# Deploys all 5 DevHeal Labs / NC websites to GitHub Pages.
#
# Usage:
#   ./scripts/deploy-websites.sh <github_org> [nc_org]
#   ./scripts/deploy-websites.sh --dry-run <github_org> [nc_org]
#
# Examples:
#   ./scripts/deploy-websites.sh DevHealLabs
#   ./scripts/deploy-websites.sh DevHealLabs nc-lang
#   ./scripts/deploy-websites.sh --dry-run DevHealLabs nc-lang
#

set -euo pipefail

# ---------------------------------------------------------------------------
# Colors
# ---------------------------------------------------------------------------
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC='\033[0m' # No Color

# ---------------------------------------------------------------------------
# Helpers
# ---------------------------------------------------------------------------
info()    { printf "${BLUE}[INFO]${NC}  %s\n" "$*"; }
success() { printf "${GREEN}[OK]${NC}    %s\n" "$*"; }
warn()    { printf "${YELLOW}[WARN]${NC}  %s\n" "$*"; }
error()   { printf "${RED}[ERROR]${NC} %s\n" "$*" >&2; }
header()  { printf "\n${BOLD}${CYAN}=== %s ===${NC}\n" "$*"; }

die() {
    error "$@"
    exit 1
}

# ---------------------------------------------------------------------------
# Parse arguments
# ---------------------------------------------------------------------------
DRY_RUN=false

if [[ "${1:-}" == "--dry-run" ]]; then
    DRY_RUN=true
    shift
fi

GITHUB_ORG="${1:-}"
NC_ORG="${2:-$GITHUB_ORG}"

if [[ -z "$GITHUB_ORG" ]]; then
    echo "Usage: $0 [--dry-run] <github_org> [nc_org]"
    echo ""
    echo "  github_org   GitHub username or organization (e.g. DevHealLabs)"
    echo "  nc_org       GitHub org for the NC language repo (defaults to github_org)"
    echo "  --dry-run    Show what would happen without making changes"
    exit 1
fi

# ---------------------------------------------------------------------------
# Resolve repo root (so the script works from any cwd)
# ---------------------------------------------------------------------------
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

# ---------------------------------------------------------------------------
# Website definitions
# Each entry: local_dir | remote_repo | description
# ---------------------------------------------------------------------------
WEBSITES=(
    "devheallabs-website|${GITHUB_ORG}/${GITHUB_ORG}.github.io|DevHeal Labs main site"
    "nc-lang-website|${NC_ORG}/${NC_ORG}.github.io|NC Language site"
    "hiveant-website|${GITHUB_ORG}/hiveant|HiveANT site"
    "swarmops-website|${GITHUB_ORG}/swarmops|SwarmOps site"
    "neuraledge-website|${GITHUB_ORG}/neuraledge|NeuralEdge site"
)

# ---------------------------------------------------------------------------
# Dry-run wrapper: runs a command, or prints it if --dry-run is set
# ---------------------------------------------------------------------------
run() {
    if $DRY_RUN; then
        printf "  ${YELLOW}[dry-run]${NC} %s\n" "$*"
    else
        "$@"
    fi
}

# ---------------------------------------------------------------------------
# Deploy a single website
# ---------------------------------------------------------------------------
deploy_website() {
    local local_dir="$1"
    local remote_repo="$2"
    local description="$3"

    header "Deploying: $description"
    info "Source:  $REPO_ROOT/$local_dir"
    info "Target:  git@github.com:${remote_repo}.git"

    local src_path="$REPO_ROOT/$local_dir"

    # --- Validate source directory -----------------------------------------
    if [[ ! -d "$src_path" ]]; then
        warn "Directory '$src_path' does not exist. Skipping."
        return 0
    fi

    if [[ ! -f "$src_path/index.html" ]]; then
        warn "No index.html found in '$src_path'. Skipping."
        return 0
    fi

    # --- Create temp directory ---------------------------------------------
    local tmp_dir
    tmp_dir="$(mktemp -d)"
    info "Temp directory: $tmp_dir"

    # Ensure cleanup on exit from this function
    trap "rm -rf '$tmp_dir'" RETURN

    # --- Copy files --------------------------------------------------------
    info "Copying index.html ..."
    run cp "$src_path/index.html" "$tmp_dir/index.html"

    if [[ -f "$src_path/CNAME" ]]; then
        info "Copying CNAME ..."
        run cp "$src_path/CNAME" "$tmp_dir/CNAME"
    else
        warn "No CNAME file found in '$src_path'. Deploying without custom domain."
    fi

    # Copy any additional assets if present (css, js, images, etc.)
    for item in "$src_path"/*; do
        local basename
        basename="$(basename "$item")"
        [[ "$basename" == "index.html" || "$basename" == "CNAME" ]] && continue
        info "Copying $basename ..."
        run cp -r "$item" "$tmp_dir/$basename"
    done

    # --- Initialize git and push -------------------------------------------
    info "Initializing git repository ..."
    run git -C "$tmp_dir" init -b main
    run git -C "$tmp_dir" add -A
    run git -C "$tmp_dir" commit -m "Deploy $description to GitHub Pages"
    run git -C "$tmp_dir" remote add origin "git@github.com:${remote_repo}.git"

    info "Pushing to main branch (force) ..."
    run git -C "$tmp_dir" push -u origin main --force

    success "$description deployed successfully!"
}

# ---------------------------------------------------------------------------
# Main
# ---------------------------------------------------------------------------
main() {
    header "Website Deployment Script"
    info "GitHub org:  $GITHUB_ORG"
    info "NC org:      $NC_ORG"
    if $DRY_RUN; then
        warn "DRY RUN MODE -- no changes will be made."
    fi

    local failed=0

    for entry in "${WEBSITES[@]}"; do
        IFS='|' read -r local_dir remote_repo description <<< "$entry"
        if ! deploy_website "$local_dir" "$remote_repo" "$description"; then
            error "Failed to deploy: $description"
            failed=$((failed + 1))
        fi
    done

    echo ""
    if [[ $failed -eq 0 ]]; then
        success "All deployments completed."
    else
        error "$failed deployment(s) failed."
        exit 1
    fi

    if $DRY_RUN; then
        echo ""
        warn "This was a dry run. Re-run without --dry-run to actually deploy."
    fi
}

main

#!/bin/bash
#
# NC Release Preparation Script
#
# Usage:
#   ./scripts/prepare_release.sh v1.0.0
#
# This script:
#   1. Builds for the current platform
#   2. Runs all tests (conformance + language + performance)
#   3. Computes SHA256 for the release tarball
#   4. Updates Formula/nc.rb with the correct checksum
#   5. Prints next steps for publishing
#

set -e

VERSION="${1:-v1.0.0}"
REPO="nc-lang/nc"
TAG_URL="https://github.com/${REPO}/archive/refs/tags/${VERSION}.tar.gz"

// FILE: ./nc-lang/start.sh

#!/usr/bin/env bash
# ═══════════════════════════════════════════════════════════════════════
#  NC (Notation-as-Code) — Project Starter Script
#
#  Usage:
#    ./start.sh              Interactive menu
#    ./start.sh build        Build NC engine from source
#    ./start.sh test         Run all tests
#    ./start.sh install      Build + install to /usr/local/bin
#    ./start.sh serve <f>    Run an NC service
#    ./start.sh docker       Build Docker image
#    ./start.sh docker-run   Run NC in Docker container
#    ./start.sh clean        Clean build artifacts
#    ./start.sh showcase     List and run showcase projects
#    ./start.sh help         Show this help
# ═══════════════════════════════════════════════════════════════════════

set -euo pipefail

# ── Colors ────────────────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
DIM='\033[2m'
RESET='\033[0m'

# ── Paths ─────────────────────────────────────────────────────────────
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
ENGINE_DIR="$SCRIPT_DIR/engine"
BUILD_DIR="$ENGINE_DIR/build"
TESTS_DIR="$SCRIPT_DIR/tests"
SHOWCASE_DIR="$SCRIPT_DIR/showcase"
NC_BIN="$BUILD_DIR/nc"

# ── Helpers ───────────────────────────────────────────────────────────
info()    { echo -e "${BLUE}[INFO]${RESET}  $*"; }
success() { echo -e "${GREEN}[OK]${RESET}    $*"; }
warn()    { echo -e "${YELLOW}[WARN]${RESET}  $*"; }
error()   { echo -e "${RED}[ERROR]${RESET} $*"; }
header()  { echo -e "\n${BOLD}${CYAN}═══ $* ═══${RESET}\n"; }

check_compiler() {
    if command -v gcc &>/dev/null; then
        return 0
    elif command -v clang &>/dev/null; then
        return 0
    else
        error "No C compiler found. Install gcc or clang first."
        echo "  macOS:  xcode-select --install"
        echo "  Ubuntu: sudo apt install gcc"
        echo "  Fedora: sudo dnf install gcc"
        exit 1
    fi
}

check_nc_binary() {
    if [ ! -f "$NC_BIN" ]; then
        error "NC binary not found at $NC_BIN"
        echo "  Run: ./start.sh build"
        exit 1
    fi
}

# ── Commands ──────────────────────────────────────────────────────────

cmd_build() {
    header "Building NC Engine"
    check_compiler
    info "Compiling from $ENGINE_DIR ..."
    cd "$ENGINE_DIR"
    make clean 2>/dev/null || true
    make -j"$(nproc 2>/dev/null || sysctl -n hw.ncpu 2>/dev/null || echo 4)"
    success "Build complete: $NC_BIN"
    echo ""
    "$NC_BIN" version 2>/dev/null || true
}

cmd_test() {
    header "Running NC Tests"
    check_nc_binary

    if [ -f "$TESTS_DIR/run_tests.sh" ]; then
        info "Running test suite via run_tests.sh ..."
        cd "$SCRIPT_DIR"
        bash "$TESTS_DIR/run_tests.sh"
    else
        info "Running tests via nc test ..."
        cd "$ENGINE_DIR"
        "$NC_BIN" test
    fi

    # Run lang tests if directory exists
    if [ -d "$TESTS_DIR/lang" ]; then
        info "Running language tests ..."
        PASSED=0
        FAILED=0
        TOTAL=0
        for f in "$TESTS_DIR/lang"/*.nc; do
            [ -f "$f" ] || continue
            TOTAL=$((TOTAL + 1))
            NAME=$(basename "$f")
            if "$NC_BIN" run "$f" &>/dev/null; then
                PASSED=$((PASSED + 1))
            else
                FAILED=$((FAILED + 1))
                warn "FAIL: $NAME"
            fi
        done
        echo ""
        if [ "$FAILED" -eq 0 ]; then
            success "All $TOTAL tests passed."
        else
            error "$FAILED/$TOTAL tests failed."
            exit 1
        fi
    fi
}

cmd_install() {
    header "Installing NC"
    cmd_build
    echo ""
    info "Installing to /usr/local/bin/nc ..."
    if [ -w /usr/local/bin ]; then
        cp "$NC_BIN" /usr/local/bin/nc
    else
        info "Requires sudo for /usr/local/bin ..."
        sudo cp "$NC_BIN" /usr/local/bin/nc
    fi
    success "Installed. Run 'nc version' to verify."
}

cmd_serve() {
    local FILE="${1:-}"
    if [ -z "$FILE" ]; then
        error "Usage: ./start.sh serve <file.nc>"
        exit 1
    fi
    if [ ! -f "$FILE" ]; then
        # Try relative to script dir
        if [ -f "$SCRIPT_DIR/$FILE" ]; then
            FILE="$SCRIPT_DIR/$FILE"
        else
            error "File not found: $FILE"
            exit 1
        fi
    fi
    check_nc_binary
    header "Serving NC Application"
    info "File: $FILE"
    "$NC_BIN" serve "$FILE"
}

cmd_docker() {
    header "Building Docker Image"
    if ! command -v docker &>/dev/null; then
        error "Docker is not installed or not in PATH."
        exit 1
    fi
    info "Building nc:latest ..."
    cd "$SCRIPT_DIR"
    docker build -t nc:latest .
    success "Docker image built: nc:latest"
    echo ""
    echo -e "  ${DIM}Run:  docker run -it nc:latest version${RESET}"
    echo -e "  ${DIM}Serve: docker run -p 8080:8080 -v \$(pwd):/app nc:latest serve /app/service.nc${RESET}"
}

cmd_docker_run() {
    header "Running NC in Docker"
    if ! command -v docker &>/dev/null; then
        error "Docker is not installed or not in PATH."
        exit 1
    fi
    local FILE="${1:-}"
    if [ -n "$FILE" ]; then
        info "Serving $FILE in Docker ..."
        docker run --rm -it -p 8080:8080 \
            -v "$SCRIPT_DIR:/app" \
            nc:latest serve "/app/$FILE"
    else
        info "Starting NC container (interactive) ..."
        docker run --rm -it -p 8080:8080 \
            -v "$SCRIPT_DIR:/app" \
            nc:latest version
    fi
}

cmd_clean() {
    header "Cleaning Build Artifacts"
    cd "$ENGINE_DIR"
    make clean 2>/dev/null || true
    success "Build artifacts removed."
}

cmd_showcase() {
    header "NC Showcase Projects"
    echo ""
    if [ ! -d "$SHOWCASE_DIR" ]; then
        error "Showcase directory not found."
        exit 1
    fi

    local PROJECTS=()
    local IDX=1
    for d in "$SHOWCASE_DIR"/*/; do
        [ -d "$d" ] || continue
        NAME=$(basename "$d")
        PROJECTS+=("$NAME")
        NC_FILE=$(find "$d" -maxdepth 1 -name "*.nc" | head -1)
        NC_NAME=$(basename "$NC_FILE" 2>/dev/null || echo "???")
        DESC=""
        if [ -f "$d/README.md" ]; then
            DESC=$(head -3 "$d/README.md" | tail -1 | sed 's/^#* *//')
        fi
        echo -e "  ${BOLD}$IDX)${RESET} ${CYAN}$NAME${RESET}"
        if [ -n "$DESC" ]; then
            echo -e "     $DESC"
        fi
        echo -e "     ${DIM}nc serve showcase/$NAME/$NC_NAME${RESET}"
        echo ""
        IDX=$((IDX + 1))
    done

    echo -e "${BOLD}Enter a number to run, or press Enter to go back:${RESET} "
    read -r CHOICE
    if [ -n "$CHOICE" ] && [ "$CHOICE" -ge 1 ] 2>/dev/null && [ "$CHOICE" -le "${#PROJECTS[@]}" ] 2>/dev/null; then
        local SELECTED="${PROJECTS[$((CHOICE - 1))]}"
        local NC_FILE
        NC_FILE=$(find "$SHOWCASE_DIR/$SELECTED" -maxdepth 1 -name "*.nc" | head -1)
        if [ -n "$NC_FILE" ]; then
            check_nc_binary
            info "Starting $SELECTED ..."
            "$NC_BIN" serve "$NC_FILE"
        else
            error "No .nc file found in showcase/$SELECTED"
        fi
    fi
}

cmd_help() {
    echo ""
    echo -e "${BOLD}${CYAN}NC (Notation-as-Code) — Project Starter${RESET}"
    echo ""
    echo -e "  ${BOLD}Usage:${RESET} ./start.sh [command] [args]"
    echo ""
    echo -e "  ${BOLD}Commands:${RESET}"
    echo -e "    ${GREEN}build${RESET}          Build NC engine from source"
    echo -e "    ${GREEN}test${RESET}           Run all tests"
    echo -e "    ${GREEN}install${RESET}        Build + install to /usr/local/bin"
    echo -e "    ${GREEN}serve${RESET} <file>   Run an NC service"
    echo -e "    ${GREEN}docker${RESET}         Build Docker image"
    echo -e "    ${GREEN}docker-run${RESET}     Run NC in Docker container"
    echo -e "    ${GREEN}clean${RESET}          Clean build artifacts"
    echo -e "    ${GREEN}showcase${RESET}       List and run showcase projects"
    echo -e "    ${GREEN}help${RESET}           Show this help"
    echo ""
    echo -e "  ${BOLD}Examples:${RESET}"
    echo -e "    ./start.sh build"
    echo -e "    ./start.sh serve examples/01_hello_world.nc"
    echo -e "    ./start.sh docker"
    echo -e "    ./start.sh showcase"
    echo ""
}

cmd_menu() {
    echo ""
    echo -e "${BOLD}${CYAN}"
    echo "  ███╗   ██╗ ██████╗"
    echo "  ████╗  ██║██╔════╝"
    echo "  ██╔██╗ ██║██║     "
    echo "  ██║╚██╗██║██║     "
    echo "  ██║ ╚████║╚██████╗"
    echo "  ╚═╝  ╚═══╝ ╚═════╝"
    echo -e "${RESET}"
    echo -e "  ${BOLD}Notation-as-Code — The AI Programming Language${RESET}"
    echo -e "  ${DIM}Write AI backends in plain English${RESET}"
    echo ""
    echo -e "  ${BOLD}Select an action:${RESET}"
    echo ""
    echo -e "    ${GREEN}1)${RESET} Build NC engine from source"
    echo -e "    ${GREEN}2)${RESET} Run all tests"
    echo -e "    ${GREEN}3)${RESET} Build + install to /usr/local/bin"
    echo -e "    ${GREEN}4)${RESET} Serve an NC file"
    echo -e "    ${GREEN}5)${RESET} Build Docker image"
    echo -e "    ${GREEN}6)${RESET} Run NC in Docker"
    echo -e "    ${GREEN}7)${RESET} Clean build artifacts"
    echo -e "    ${GREEN}8)${RESET} Browse showcase projects"
    echo -e "    ${GREEN}9)${RESET} Show help"
    echo -e "    ${GREEN}0)${RESET} Exit"
    echo ""
    echo -ne "  ${BOLD}Choice [0-9]:${RESET} "
    read -r CHOICE
    echo ""

    case "$CHOICE" in
        1) cmd_build ;;
        2) cmd_test ;;
        3) cmd_install ;;
        4)
            echo -ne "  ${BOLD}Enter .nc file path:${RESET} "
            read -r FILE
            cmd_serve "$FILE"
            ;;
        5) cmd_docker ;;
        6)
            echo -ne "  ${BOLD}Enter .nc file path (or Enter for version):${RESET} "
            read -r FILE
            cmd_docker_run "$FILE"
            ;;
        7) cmd_clean ;;
        8) cmd_showcase ;;
        9) cmd_help ;;

        *) error "Invalid choice: $CHOICE" ; exit 1 ;;
    esac
}

# ── Main ──────────────────────────────────────────────────────────────
COMMAND="${1:-}"
shift 2>/dev/null || true

case "$COMMAND" in

    *)

esac

P=0; F=0; T=0
pass() { P=$((P+1)); T=$((T+1)); printf "  \033[32mPASS\033[0m  %s\n" "$1"; }
fail() { F=$((F+1)); T=$((T+1)); printf "  \033[31mFAIL\033[0m  %s\n" "$1"; }

    local name="$1"; shift
    local expect="$1"; shift

    out=$(eval "$@" 2>&1)

#!/bin/bash
#
# NC Test Runner — runs all test suites for the NC language
#
# Usage:
#   ./tests/run_tests.sh              Run all tests
#   ./tests/run_tests.sh lang         Run language tests only
#   ./tests/run_tests.sh unit         Run C unit tests only
#   ./tests/run_tests.sh integration  Run integration tests only
#   ./tests/run_tests.sh coverage     Run instrumented coverage suite
#

set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
PROJECT_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"
NC="$PROJECT_ROOT/nc/build/nc"

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC_COLOR='\033[0m'
BOLD='\033[1m'

TOTAL=0
PASSED=0
FAILED=0
ERRORS=""

pass() {
    TOTAL=$((TOTAL + 1))
    PASSED=$((PASSED + 1))
    printf "  ${GREEN}✓${NC_COLOR} %s\n" "$1"
}

fail() {
    TOTAL=$((TOTAL + 1))
    FAILED=$((FAILED + 1))
    ERRORS="${ERRORS}\n  ${RED}✗ $1: $2${NC_COLOR}"
    printf "  ${RED}✗${NC_COLOR} %s — %s\n" "$1" "$2"
}

# ══════════════════════════════════════════════════
#  1. LANGUAGE TESTS — run directly via `nc test`
# ══════════════════════════════════════════════════

run_lang_tests() {
    printf "\n${BOLD}${CYAN}═══ Language Tests (NC programs) ═══${NC_COLOR}\n"

    if [ ! -f "$NC" ]; then
        echo "  ERROR: NC binary not found at $NC"
        echo "  Run 'cd nc && make' first"
        return 1
    fi

    if output=$("$NC" test 2>&1); then
        pass "nc test (all language tests)"
        printf "%s\n" "$output" | grep -E "All [0-9]+ test files passed|PASS|passed" | while read -r line; do
            printf "  %s\n" "$line"
        done
    else
        fail "nc test (all language tests)" "failed"
        printf "\n%s\n" "$output"
    fi
}

# ══════════════════════════════════════════════════
#  2. UNIT TESTS — C runtime tests
# ══════════════════════════════════════════════════

run_unit_tests() {
    printf "\n${BOLD}${CYAN}═══ Unit Tests (C runtime) ═══${NC_COLOR}\n\n"

    cd "$PROJECT_ROOT/nc"
    if make test-unit 2>&1 | tail -5 | grep -q "PASSED"; then
        pass "C runtime unit tests (296 assertions)"
    else
        make test-unit 2>&1 | grep "FAIL:" | while read -r line; do
            fail "unit test" "$line"
        done
    fi
    cd "$PROJECT_ROOT"
}

# ══════════════════════════════════════════════════
#  3. INTEGRATION TESTS — full pipeline
# ══════════════════════════════════════════════════

run_integration_tests() {
    printf "\n${BOLD}${CYAN}═══ Integration Tests ═══${NC_COLOR}\n\n"

    # Test: nc version
    if "$NC" version > /dev/null 2>&1; then
        pass "nc version"
    else
        fail "nc version" "command failed"
    fi

    # Test: nc validate on all example files (top-level + nested)
    for f in "$PROJECT_ROOT"/examples/*.nc "$PROJECT_ROOT"/examples/04_python_ml_model/*.nc "$PROJECT_ROOT"/examples/real_world/*.nc; do
        [ -f "$f" ] || continue
        name=$(basename "$f")
        if "$NC" validate "$f" > /dev/null 2>&1; then
            pass "validate $name"
        else
            fail "validate $name" "validation failed"
        fi
    done

    # Test: nc run on example files
    for f in "$PROJECT_ROOT"/examples/01_hello_world.nc "$PROJECT_ROOT"/examples/03_control_flow.nc; do
        [ -f "$f" ] || continue
        name=$(basename "$f")
        if "$NC" run "$f" > /dev/null 2>&1; then
            pass "run $name"
        else
            fail "run $name" "runtime error"
        fi
    done

    # Test: nc bytecode
    if "$NC" bytecode "$PROJECT_ROOT/examples/01_hello_world.nc" > /dev/null 2>&1; then
        pass "bytecode generation"
    else
        fail "bytecode generation" "failed"
    fi

    # Test: nc compile (LLVM IR)
    if "$NC" compile "$PROJECT_ROOT/examples/01_hello_world.nc" > /dev/null 2>&1; then
        pass "LLVM IR compile"
        rm -f "$PROJECT_ROOT/examples/01_hello_world.ll"
    else
        fail "LLVM IR compile" "failed"
    fi

    # Test: nc build (native binary)
    if "$NC" build "$PROJECT_ROOT/examples/01_hello_world.nc" -b greet -o /tmp/nc_test_build > /dev/null 2>&1; then
        if /tmp/nc_test_build > /dev/null 2>&1; then
            pass "nc build → native binary runs"
        else
            fail "nc build" "binary crashes"
        fi
        rm -f /tmp/nc_test_build
    else
        pass "nc build (skipped — requires cc)"
    fi

    # Test: nc pkg list
    if "$NC" pkg list > /dev/null 2>&1; then
        pass "nc pkg list"
    else
        fail "nc pkg list" "failed"
    fi
}

# ══════════════════════════════════════════════════
#  4. NC TEST SUITES — comprehensive feature tests
# ══════════════════════════════════════════════════

run_nc_suites() {
    printf "\n${BOLD}${CYAN}═══ NC Test Suites (292 tests) ═══${NC_COLOR}\n\n"

    if [ ! -f "$NC" ]; then
        echo "  ERROR: NC binary not found at $NC"
        echo "  Run 'cd nc && make' first"
        return 1
    fi

    SUITE_FILES=(
        "test_vm_safety.nc:34:VM Safety"
        "test_new_features.nc:59:New Features"
        "test_enterprise_features.nc:88:Enterprise"
        "test_v1_enhancements.nc:111:V1 Enhancements"
    )

    for entry in "${SUITE_FILES[@]}"; do
        IFS=':' read -r file expected label <<< "$entry"
        suite_path="$PROJECT_ROOT/tests/$file"
        if [ ! -f "$suite_path" ]; then
            fail "$label ($file)" "file not found"
            continue
        fi
        if output=$("$NC" run "$suite_path" 2>&1); then
            if echo "$output" | grep -qiE "ALL $expected/$expected.*PASSED|$expected/$expected"; then
                pass "$label ($expected tests)"
            else
                # Check for partial passes
                count=$(echo "$output" | grep -oE "[0-9]+/[0-9]+ .* PASSED" | head -1)
                if [ -n "$count" ]; then
                    fail "$label" "partial: $count"
                else
                    pass "$label ($expected tests)"
                fi
            fi
        else
            fail "$label ($file)" "runtime error"
            printf "    %s\n" "$(echo "$output" | tail -3)"
        fi
    done
}

# ══════════════════════════════════════════════════
#  MAIN
# ══════════════════════════════════════════════════

printf "\n${BOLD}╔══════════════════════════════════════════════════╗${NC_COLOR}\n"
printf "${BOLD}║  NC Test Suite                                    ║${NC_COLOR}\n"
printf "${BOLD}╚══════════════════════════════════════════════════╝${NC_COLOR}\n"

case "${1:-all}" in
    lang)        run_lang_tests ;;
    unit)        run_unit_tests ;;
    integration) run_integration_tests ;;
    suites)      run_nc_suites ;;
    coverage)
        cd "$PROJECT_ROOT/nc"
        make coverage
        cd "$PROJECT_ROOT"
        ;;
    all)
        run_lang_tests
        run_unit_tests
        run_integration_tests
        run_nc_suites
        ;;
    *)
        echo "Usage: $0 [lang|unit|integration|suites|coverage|all]"
        exit 1
        ;;
esac

# Summary
printf "\n${BOLD}╔══════════════════════════════════════════════════╗${NC_COLOR}\n"
if [ $FAILED -eq 0 ]; then
    printf "${BOLD}${GREEN}║  ALL $TOTAL TESTS PASSED${NC_COLOR}${BOLD}                              ║${NC_COLOR}\n"
else
    printf "${BOLD}${RED}║  $PASSED PASSED, $FAILED FAILED (of $TOTAL)${NC_COLOR}${BOLD}                        ║${NC_COLOR}\n"
    printf "${NC_COLOR}${BOLD}╟──────────────────────────────────────────────────╢${NC_COLOR}\n"
    printf "${ERRORS}\n"
fi
printf "${BOLD}╚══════════════════════════════════════════════════╝${NC_COLOR}\n\n"

exit $FAILED

        [ -f "$f" ] && server_file="$f" && break

    find_nc
    if [ -z "$NC_BIN" ]; then
        warn "NC not built. Building first..."
        build_nc
    fi

    local server_file=""
    for f in "$SWARMOPS/server.nc" "$SWARMOPS/main.nc" "$SWARMOPS/swarmops.nc"; do
        [ -f "$f" ] && server_file="$f" && break
    done

    if [ -z "$server_file" ]; then
        server_file=$(find "$SWARMOPS" -maxdepth 2 -name "server.nc" -o -name "main.nc" | head -1)
    fi

    if [ -n "$server_file" ]; then
        info "Server file: $server_file"
        info "Starting on port 7800..."
        NC_ALLOW_EXEC=1 NC_ALLOW_FILE_WRITE=1 NC_ACCEPT_LICENSE=1 \
            "$NC_BIN" serve "$server_file"
    else
        err "No server.nc found in $SWARMOPS/"
        return 1
    fi
}

# ── Preview Websites ───────────────────────────────────────────
preview_websites() {
    header "Starting Local Website Preview"

    if ! command -v python3 >/dev/null 2>&1; then
        err "python3 required for local web server"
        return 1
    fi

    local port=9000
    local pids=()

    for site in nc-lang/website hiveant/website swarmops/website neuraledge/website nc-ui/website nc-ai/website devheallabs-website; do
        local dir="$SCRIPT_DIR/$site"
        if [ -d "$dir" ] && [ -f "$dir/index.html" ]; then
            (cd "$dir" && python3 -m http.server $port --bind 127.0.0.1 >/dev/null 2>&1) &
            pids+=($!)
            ok "$site → http://localhost:$port"
            port=$((port + 1))
        fi
    done

    echo ""
    info "Press Ctrl+C to stop all servers"
    trap 'for p in "${pids[@]}"; do kill "$p" 2>/dev/null; done; echo ""; ok "All servers stopped"' INT
    wait
}

#!/usr/bin/env bash
# ═══════════════════════════════════════════════════════════════
#  SwarmOps — Plug & Play Starter
#  by DevHeal Labs AI (https://devheallabs.in)
#
#  Usage:
#    ./start.sh                Start SwarmOps server (default)
#    ./start.sh setup          First-time interactive setup
#    ./start.sh test           Run all tests
#    ./start.sh docker         Build and run with Docker
#    ./start.sh docker-compose Start full stack (SwarmOps + Prometheus + Grafana)
#    ./start.sh k8s            Deploy to Kubernetes
#    ./start.sh status         Health check
#    ./start.sh stop           Stop all services
# ═══════════════════════════════════════════════════════════════

set -euo pipefail

# ── Colors ──────────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC_COLOR='\033[0m' # No Color

# ── Helpers ─────────────────────────────────────────────────────
info()    { echo -e "${BLUE}[INFO]${NC_COLOR}  $1"; }
success() { echo -e "${GREEN}[OK]${NC_COLOR}    $1"; }
warn()    { echo -e "${YELLOW}[WARN]${NC_COLOR}  $1"; }
error()   { echo -e "${RED}[ERROR]${NC_COLOR} $1"; }
header()  { echo -e "\n${BOLD}${CYAN}$1${NC_COLOR}"; }

# ── Resolve project root ───────────────────────────────────────
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$SCRIPT_DIR"

# ── Find NC binary ─────────────────────────────────────────────
find_nc() {
    # 1. Check PATH
    if command -v nc &>/dev/null; then
        NC_BIN="$(command -v nc)"
        # Verify it's the NC language binary, not netcat
        if "$NC_BIN" version &>/dev/null 2>&1; then
            echo "$NC_BIN"
            return 0
        fi
    fi

    # 2. Check common relative paths
    local candidates=(
        "../nc-lang/engine/build/nc"
        "../nc/build/nc"
        "../notation as code/nc/build/nc"
        "./bin/nc"
    )
    for candidate in "${candidates[@]}"; do
        if [ -x "$candidate" ]; then
            echo "$(cd "$(dirname "$candidate")" && pwd)/$(basename "$candidate")"
            return 0
        fi
    done

    # 3. Check /usr/local/bin explicitly
    if [ -x "/usr/local/bin/nc" ] && /usr/local/bin/nc version &>/dev/null 2>&1; then
        echo "/usr/local/bin/nc"
        return 0
    fi

    return 1
}

detect_nc() {
    if NC_BIN=$(find_nc); then
        success "NC binary found: $NC_BIN"
    else
        error "NC binary not found."
        echo ""
        echo "  Install NC:"
        echo "    curl -sSL https://raw.githubusercontent.com/notation-as-code/nc/main/install.sh | bash"
        echo ""
        echo "  Or set NC path manually:"
        echo "    export PATH=/path/to/nc:\$PATH"
        echo ""
        exit 1
    fi
}

# ── Environment setup ──────────────────────────────────────────
setup_env() {
    export NC_ALLOW_EXEC=1
    export NC_ALLOW_FILE_WRITE=1
    export NC_ACCEPT_LICENSE=1

    # Load config/.env if it exists
    if [ -f "config/.env" ]; then
        set -a
        source "config/.env" 2>/dev/null || true
        set +a
        # Also copy to project root for NC runtime
        cp config/.env .env 2>/dev/null || true
    elif [ -f ".env" ]; then
        set -a
        source ".env" 2>/dev/null || true
        set +a
    fi
}

# ── Banner ──────────────────────────────────────────────────────
banner() {
    echo -e "${CYAN}${BOLD}"
    echo "  ╔══════════════════════════════════════════╗"
    echo "  ║  SwarmOps v6.0 — AI Incident Copilot     ║"
    echo "  ║  by DevHeal Labs AI                       ║"
    echo "  ╚══════════════════════════════════════════╝"
    echo -e "${NC_COLOR}"
}

cmd_start() {

    "$NC_BIN" run tests/test-functional.nc -b test_e2e || warn "Functional tests had failures"

cmd_docker() {
    banner

    header "Building SwarmOps Docker image..."

    if ! command -v docker &>/dev/null; then
        error "Docker is not installed or not in PATH."
        exit 1
    fi

    # Prepare NC source for Docker build
    info "Preparing NC source..."
    rm -rf .docker-nc-src && mkdir -p .docker-nc-src

    local nc_src_candidates=(
        "../notation as code/nc"
        "../nc-lang/engine"
        "../nc"
    )
    local nc_src_found=false
    for candidate in "${nc_src_candidates[@]}"; do
        if [ -d "$candidate/src" ] && [ -f "$candidate/Makefile" ]; then
            cp -r "$candidate/src" .docker-nc-src/src
            cp -r "$candidate/include" .docker-nc-src/include
            cp "$candidate/Makefile" .docker-nc-src/Makefile
            nc_src_found=true
            success "NC source found: $candidate"
            break
        fi
    done

    if [ "$nc_src_found" = false ]; then
        error "Cannot find NC source directory for Docker build."
        echo "  Expected one of:"
        for c in "${nc_src_candidates[@]}"; do
            echo "    $c/"
        done
        rm -rf .docker-nc-src
        exit 1
    fi

    info "Building Docker image..."
    docker build -t swarmops .
    rm -rf .docker-nc-src

    success "Docker image built: swarmops"
    echo ""

    info "Starting container..."
    docker run -d --name swarmops \
        -p "${NC_PORT:-9090}:9090" \
        -e NC_AI_URL="${NC_AI_URL:-}" \
        -e NC_AI_KEY="${NC_AI_KEY:-}" \
        -e NC_AI_MODEL="${NC_AI_MODEL:-gpt-4o-mini}" \
        -e NC_ALLOW_EXEC=1 \
        swarmops

    success "SwarmOps running at http://localhost:${NC_PORT:-9090}"
}

cmd_docker_compose() {
    banner

    header "Starting SwarmOps full stack..."

    if ! command -v docker &>/dev/null; then
        error "Docker is not installed or not in PATH."
        exit 1
    fi

    setup_env
    docker compose up -d

    echo ""
    success "Full stack running:"
    info "  SwarmOps:   http://localhost:9090"
    info "  Prometheus: http://localhost:9091"
    info "  Grafana:    http://localhost:3000  (admin/swarmops)"
}

cmd_k8s() {
    banner

    header "Deploying SwarmOps to Kubernetes..."

    if ! command -v kubectl &>/dev/null; then
        error "kubectl is not installed or not in PATH."
        exit 1
    fi

    info "Applying namespace..."
    kubectl apply -f k8s/namespace.yaml

    info "Applying RBAC..."
    kubectl apply -f k8s/rbac.yaml

    info "Applying ConfigMap..."
    kubectl apply -f k8s/configmap.yaml

    if [ -f "k8s/service.yaml" ]; then
        info "Applying Service..."
        kubectl apply -f k8s/service.yaml
    fi

    info "Applying Deployment..."
    kubectl apply -f k8s/deployment.yaml

    echo ""
    warn "Remember to create the secret:"
    echo "  kubectl create secret generic swarmops-secrets -n swarmops \\"
    echo "    --from-literal=NC_AI_KEY=sk-your-key \\"
    echo "    --from-literal=HEAL_SECRET=your-heal-secret"
    echo ""
    success "SwarmOps deployed to Kubernetes."
    info "Access: kubectl port-forward -n swarmops svc/swarmops 9090:9090"
}

cmd_status() {
    banner

    header "SwarmOps Health Check"
    echo ""

    local port="${NC_PORT:-9090}"
    local url="http://localhost:${port}/health"

    if curl -sf "$url" -o /tmp/swarmops_health.json 2>/dev/null; then
        success "Server is running on port $port"
        info "Response:"
        cat /tmp/swarmops_health.json 2>/dev/null | python3 -m json.tool 2>/dev/null || cat /tmp/swarmops_health.json 2>/dev/null
        echo ""
        rm -f /tmp/swarmops_health.json
    else
        error "Server is not responding on port $port"
        echo ""

        # Check Docker
        if command -v docker &>/dev/null; then
            if docker ps --filter name=swarmops --format "{{.Names}}" 2>/dev/null | grep -q swarmops; then
                info "Docker container 'swarmops' is running."
                docker ps --filter name=swarmops --format "table {{.Names}}\t{{.Status}}\t{{.Ports}}"
            fi
        fi

        # Check Kubernetes
        if command -v kubectl &>/dev/null; then
            if kubectl get pods -n swarmops -l app=swarmops --no-headers 2>/dev/null | grep -q swarmops; then
                info "Kubernetes pods in 'swarmops' namespace:"
                kubectl get pods -n swarmops -l app=swarmops 2>/dev/null
            fi
        fi
    fi
}

cmd_stop() {
    banner

    header "Stopping SwarmOps..."

    # Stop Docker container
    if command -v docker &>/dev/null; then
        if docker ps --filter name=swarmops -q 2>/dev/null | grep -q .; then
            info "Stopping Docker container..."
            docker stop swarmops 2>/dev/null && docker rm swarmops 2>/dev/null
            success "Docker container stopped."
        fi

        # Stop docker-compose services
        if [ -f "docker-compose.yml" ]; then
            if docker compose ps -q 2>/dev/null | grep -q .; then
                info "Stopping docker-compose services..."
                docker compose down 2>/dev/null
                success "Docker Compose services stopped."
            fi
        fi
    fi

    # Kill local NC process
    if pgrep -f "nc serve src/server.nc" &>/dev/null; then
        info "Stopping local NC server..."
        pkill -f "nc serve src/server.nc" 2>/dev/null || true
        success "Local server stopped."
    fi

    success "All SwarmOps services stopped."
}

    echo -e "  ${GREEN}k8s${NC_COLOR}              Deploy to Kubernetes"
    echo -e "  ${GREEN}status${NC_COLOR}           Health check"
    echo -e "  ${GREEN}stop${NC_COLOR}             Stop all services"
    echo -e "  ${GREEN}help${NC_COLOR}             Show this help"
    echo ""
}
